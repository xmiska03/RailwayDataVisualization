196// API to read values from a single column. This is a main client facing API.
197template <typename DType>
198class TypedColumnReader : public ColumnReader {
199 public:
200 using T = typename DType::c_type;
201
202 // Read a batch of repetition levels, definition levels, and values from the
203 // column.
204 //
205 // Since null values are not stored in the values, the number of values read
206 // may be less than the number of repetition and definition levels. With
207 // nested data this is almost certainly true.
208 //
209 // Set def_levels or rep_levels to nullptr if you want to skip reading them.
210 // This is only safe if you know through some other source that there are no
211 // undefined values.
212 //
213 // To fully exhaust a row group, you must read batches until the number of
214 // values read reaches the number of stored values according to the metadata.
215 //
216 // This API is the same for both V1 and V2 of the DataPage
217 //
218 // @returns: actual number of levels read (see values_read for number of values read)
219 virtual int64_t ReadBatch(int64_t batch_size, int16_t* def_levels, int16_t* rep_levels,
220 T* values, int64_t* values_read) = 0;
221
222 // Skip reading values. This method will work for both repeated and
223 // non-repeated fields. Note that this method is skipping values and not
224 // records. This distinction is important for repeated fields, meaning that
225 // we are not skipping over the values to the next record. For example,
226 // consider the following two consecutive records containing one repeated field:
227 // {[1, 2, 3]}, {[4, 5]}. If we Skip(2), our next read value will be 3, which
228 // is inside the first record.
229 // Returns the number of values skipped.
230 virtual int64_t Skip(int64_t num_values_to_skip) = 0;
231
232 // Read a batch of repetition levels, definition levels, and indices from the
233 // column. And read the dictionary if a dictionary page is encountered during
234 // reading pages. This API is similar to ReadBatch(), with ability to read
235 // dictionary and indices. It is only valid to call this method when the reader can
236 // expose dictionary encoding. (i.e., the reader's GetExposedEncoding() returns
237 // DICTIONARY).
238 //
239 // The dictionary is read along with the data page. When there's no data page,
240 // the dictionary won't be returned.
241 //
242 // @param batch_size The batch size to read
243 // @param[out] def_levels The Parquet definition levels.
244 // @param[out] rep_levels The Parquet repetition levels.
245 // @param[out] indices The dictionary indices.
246 // @param[out] indices_read The number of indices read.
247 // @param[out] dict The pointer to dictionary values. It will return nullptr if
248 // there's no data page. Each column chunk only has one dictionary page. The dictionary
249 // is owned by the reader, so the caller is responsible for copying the dictionary
250 // values before the reader gets destroyed.
251 // @param[out] dict_len The dictionary length. It will return 0 if there's no data
252 // page.
253 // @returns: actual number of levels read (see indices_read for number of
254 // indices read
255 //
256 // \note API EXPERIMENTAL
257 virtual int64_t ReadBatchWithDictionary(int64_t batch_size, int16_t* def_levels,
258 int16_t* rep_levels, int32_t* indices,
259 int64_t* indices_read, const T** dict,
260 int32_t* dict_len) = 0;
261};
