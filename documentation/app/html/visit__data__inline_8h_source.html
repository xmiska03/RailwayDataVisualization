38// Numeric and primitive C-compatible types
39template <typename T>
40struct ArraySpanInlineVisitor<T, enable_if_has_c_type<T>> {
41 using c_type = typename T::c_type;
42
43 template <typename ValidFunc, typename NullFunc>
44 static Status VisitStatus(const ArraySpan& arr, ValidFunc&& valid_func,
45 NullFunc&& null_func) {
46 if constexpr (std::is_same_v<T, BooleanType>) {
47 int64_t offset = arr.offset;
48 const uint8_t* data = arr.buffers[1].data;
49 return VisitBitBlocks(
50 arr.buffers[0].data, offset, arr.length,
51 [&](int64_t i) { return valid_func(bit_util::GetBit(data, offset + i)); },
52 std::forward<NullFunc>(null_func));
53 } else {
54 const c_type* data = arr.GetValues<c_type>(1);
55 auto visit_valid = [&](int64_t i) { return valid_func(data[i]); };
56 return VisitBitBlocks(arr.buffers[0].data, arr.offset, arr.length,
57 std::move(visit_valid), std::forward<NullFunc>(null_func));
58 }
59 }
60
61 template <typename ValidFunc, typename NullFunc>
62 static void VisitVoid(const ArraySpan& arr, ValidFunc&& valid_func,
63 NullFunc&& null_func) {
64 if constexpr (std::is_same_v<T, BooleanType>) {
65 int64_t offset = arr.offset;
66 const uint8_t* data = arr.buffers[1].data;
67 VisitBitBlocksVoid(
68 arr.buffers[0].data, offset, arr.length,
69 [&](int64_t i) { valid_func(bit_util::GetBit(data, offset + i)); },
70 std::forward<NullFunc>(null_func));
71 } else {
72 const c_type* data = arr.GetValues<c_type>(1);
73 auto visit_valid = [&](int64_t i) { valid_func(data[i]); };
74 VisitBitBlocksVoid(arr.buffers[0].data, arr.offset, arr.length,
75 std::move(visit_valid), std::forward<NullFunc>(null_func));
76 }
77 }
78};

80// Binary, String...
81template <typename T>
82struct ArraySpanInlineVisitor<T, enable_if_base_binary<T>> {
83 using c_type = std::string_view;
84
85 template <typename ValidFunc, typename NullFunc>
86 static Status VisitStatus(const ArraySpan& arr, ValidFunc&& valid_func,
87 NullFunc&& null_func) {
88 using offset_type = typename T::offset_type;
89 constexpr char empty_value = 0;
90
91 if (arr.length == 0) {
92 return Status::OK();
93 }
94 const offset_type* offsets = arr.GetValues<offset_type>(1);
95 const char* data;
96 if (arr.buffers[2].data == NULLPTR) {
97 data = &empty_value;
98 } else {
99 // Do not apply the array offset to the values array; the value_offsets
100 // index the non-sliced values array.
101 data = arr.GetValues<char>(2, /*absolute_offset=*/0);
102 }
103 offset_type cur_offset = *offsets++;
104 return VisitBitBlocks(
105 arr.buffers[0].data, arr.offset, arr.length,
106 [&](int64_t i) {
107 ARROW_UNUSED(i);
108 auto value = std::string_view(data + cur_offset, *offsets - cur_offset);
109 cur_offset = *offsets++;
110 return valid_func(value);
111 },
112 [&]() {
113 cur_offset = *offsets++;
114 return null_func();
115 });
116 }
117
118 template <typename ValidFunc, typename NullFunc>
119 static void VisitVoid(const ArraySpan& arr, ValidFunc&& valid_func,
120 NullFunc&& null_func) {
121 using offset_type = typename T::offset_type;
122 constexpr uint8_t empty_value = 0;
123
124 if (arr.length == 0) {
125 return;
126 }
127 const offset_type* offsets = arr.GetValues<offset_type>(1);
128 const uint8_t* data;
129 if (arr.buffers[2].data == NULLPTR) {
130 data = &empty_value;
131 } else {
132 // Do not apply the array offset to the values array; the value_offsets
133 // index the non-sliced values array.
134 data = arr.GetValues<uint8_t>(2, /*absolute_offset=*/0);
135 }
136
137 VisitBitBlocksVoid(
138 arr.buffers[0].data, arr.offset, arr.length,
139 [&](int64_t i) {
140 auto value = std::string_view(reinterpret_cast<const char*>(data + offsets[i]),
141 offsets[i + 1] - offsets[i]);
142 valid_func(value);
143 },
144 std::forward<NullFunc>(null_func));
145 }
146};

193 const char* data = arr.GetValues<char>(1,

195 return VisitBitBlocks(
196 arr.buffers[0].data, arr.offset, arr.length,
197 [&](int64_t i) {
198 auto value = std::string_view(data, byte_width);

200 return valid_func(value);
201 },
202 [&]() {
203 data += byte_width;
204 return null_func();
205 });
206 }

243// Visit an array's data values, in order, without overhead.
244//
245// The Visit method's `visitor` argument should be an object with two public methods:
246// - Status VisitNull()
247// - Status VisitValue(<scalar>)
248//
249// The scalar value's type depends on the array data type:
250// - the type's `c_type`, if any
251// - for boolean arrays, a `bool`
252// - for binary, string, large binary and string, binary and string view, and fixed-size
253// binary arrays, a `std::string_view`
254
255template <typename T>
256struct ArraySpanVisitor {
257 using InlineVisitorType = internal::ArraySpanInlineVisitor<T>;
258 using c_type = typename InlineVisitorType::c_type;
259
260 template <typename Visitor>
261 static Status Visit(const ArraySpan& arr, Visitor* visitor) {
262 return InlineVisitorType::VisitStatus(
263 arr, [visitor](c_type v) { return visitor->VisitValue(v); },
264 [visitor]() { return visitor->VisitNull(); });
265 }
266};

268// Visit a null bitmap, in order, without overhead.
269//
270// The given `ValidFunc` should be a callable with either of these signatures:
271// - void()
272// - Status()
273//
274// The `NullFunc` should have the same return type as `ValidFunc`.
275
276template <typename ValidFunc, typename NullFunc>
277typename internal::call_traits::enable_if_return<ValidFunc, Status>::type
278VisitNullBitmapInline(const uint8_t* valid_bits, int64_t valid_bits_offset,
279 int64_t num_values, int64_t null_count, ValidFunc&& valid_func,
280 NullFunc&& null_func) {
281 internal::OptionalBitBlockCounter bit_counter(null_count == 0 ? NULLPTR : valid_bits,
282 valid_bits_offset, num_values);
283 int64_t position = 0;
284 int64_t offset_position = valid_bits_offset;
285 while (position < num_values) {
286 internal::BitBlockCount block = bit_counter.NextBlock();
287 if (block.AllSet()) {
288 for (int64_t i = 0; i < block.length; ++i) {
289 ARROW_RETURN_NOT_OK(valid_func());
290 }
291 } else if (block.NoneSet()) {
292 for (int64_t i = 0; i < block.length; ++i) {
293 ARROW_RETURN_NOT_OK(null_func());
294 }
295 } else {
296 for (int64_t i = 0; i < block.length; ++i) {
297 ARROW_RETURN_NOT_OK(bit_util::GetBit(valid_bits, offset_position + i)
298 ? valid_func()
299 : null_func());
300 }
301 }
302 position += block.length;
303 offset_position += block.length;
304 }
305 return Status::OK();
306}
