318class ARROW_EXPORT CollectListener : public Listener {
319 public:
320 CollectListener() : schema_(), filtered_schema_(), record_batches_(), metadatas_() {}
321 virtual ~CollectListener() = default;
322
323 Status OnSchemaDecoded(std::shared_ptr<Schema> schema,
324 std::shared_ptr<Schema> filtered_schema) override {
325 schema_ = std::move(schema);
326 filtered_schema_ = std::move(filtered_schema);
327 return Status::OK();
328 }
329
330 Status OnRecordBatchWithMetadataDecoded(
331 RecordBatchWithMetadata record_batch_with_metadata) override {
332 record_batches_.push_back(std::move(record_batch_with_metadata.batch));
333 metadatas_.push_back(std::move(record_batch_with_metadata.custom_metadata));
334 return Status::OK();
335 }
336
338 std::shared_ptr<Schema> schema() const { return schema_; }
339
341 std::shared_ptr<Schema> filtered_schema() const { return filtered_schema_; }
342
344 const std::vector<std::shared_ptr<RecordBatch>>& record_batches() const {
345 return record_batches_;
346 }
347
349 const std::vector<std::shared_ptr<KeyValueMetadata>>& metadatas() const {
350 return metadatas_;
351 }
352
354 int64_t num_record_batches() const { return record_batches_.size(); }
355
358 std::shared_ptr<RecordBatch> PopRecordBatch() {
359 auto record_batch_with_metadata = PopRecordBatchWithMetadata();
360 return std::move(record_batch_with_metadata.batch);
361 }
362
365 RecordBatchWithMetadata PopRecordBatchWithMetadata() {
366 RecordBatchWithMetadata record_batch_with_metadata;
367 if (record_batches_.empty()) {
368 return record_batch_with_metadata;
369 }
370 record_batch_with_metadata.batch = std::move(record_batches_.back());
371 record_batch_with_metadata.custom_metadata = std::move(metadatas_.back());
372 record_batches_.pop_back();
373 metadatas_.pop_back();
374 return record_batch_with_metadata;
375 }
376
377 private:
378 std::shared_ptr<Schema> schema_;
379 std::shared_ptr<Schema> filtered_schema_;
380 std::vector<std::shared_ptr<RecordBatch>> record_batches_;
381 std::vector<std::shared_ptr<KeyValueMetadata>> metadatas_;
382};
