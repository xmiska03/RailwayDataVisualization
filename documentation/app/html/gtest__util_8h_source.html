51#define ASSERT_RAISES(ENUM, expr) \
52 for (::arrow::Status _st = ::arrow::internal::GenericToStatus((expr)); \
53 !_st.Is##ENUM();) \
54 FAIL() << "Expected '" ARROW_STRINGIFY(expr) "' to fail with " ARROW_STRINGIFY( \
55 ENUM) ", but got " \
56 << _st.ToString()
57
58#define ASSERT_RAISES_WITH_MESSAGE(ENUM, message, expr) \
59 do { \
60 auto _res = (expr); \
61 ::arrow::Status _st = ::arrow::internal::GenericToStatus(_res); \
62 if (!_st.Is##ENUM()) { \
63 FAIL() << "Expected '" ARROW_STRINGIFY(expr) "' to fail with " ARROW_STRINGIFY( \
64 ENUM) ", but got " \
65 << _st.ToString(); \
66 } \
67 ASSERT_EQ((message), _st.ToStringWithoutContextLines()); \
68 } while (false)
69
70#define EXPECT_RAISES_WITH_MESSAGE_THAT(ENUM, matcher, expr) \
71 do { \
72 auto _res = (expr); \
73 ::arrow::Status _st = ::arrow::internal::GenericToStatus(_res); \
74 EXPECT_TRUE(_st.Is##ENUM()) << "Expected '" ARROW_STRINGIFY(expr) "' to fail with " \
75 << ARROW_STRINGIFY(ENUM) ", but got " << _st.ToString(); \
76 EXPECT_THAT(_st.ToStringWithoutContextLines(), (matcher)); \
77 } while (false)
78
79#define EXPECT_RAISES_WITH_CODE_AND_MESSAGE_THAT(code, matcher, expr) \
80 do { \
81 auto _res = (expr); \
82 ::arrow::Status _st = ::arrow::internal::GenericToStatus(_res); \
83 EXPECT_EQ(_st.CodeAsString(), Status::CodeAsString(code)); \
84 EXPECT_THAT(_st.ToStringWithoutContextLines(), (matcher)); \
85 } while (false)

479// A data type with only move constructors (no copy, no default).
480struct MoveOnlyDataType {
481 explicit MoveOnlyDataType(int x) : data(new int(x)) {}
482
483 MoveOnlyDataType(const MoveOnlyDataType& other) = delete;
484 MoveOnlyDataType& operator=(const MoveOnlyDataType& other) = delete;
485
486 MoveOnlyDataType(MoveOnlyDataType&& other) { MoveFrom(&other); }
487 MoveOnlyDataType& operator=(MoveOnlyDataType&& other) {
488 MoveFrom(&other);
489 return *this;
490 }
491
492 MoveOnlyDataType& operator=(int x) {
493 if (data != nullptr) {
494 delete data;
495 }
496 data = new int(x);
497 return *this;
498 }
499
500 ~MoveOnlyDataType() { Destroy(); }
501
502 void Destroy() {
503 if (data != nullptr) {
504 delete data;
505 data = nullptr;
506 moves = -1;
507 }
508 }
509
510 void MoveFrom(MoveOnlyDataType* other) {
511 Destroy();
512 data = other->data;
513 other->data = nullptr;
514 moves = other->moves + 1;
515 }
516
517 int ToInt() const { return data == nullptr ? -42 : *data; }
518
519 bool operator==(const MoveOnlyDataType& other) const {
520 return data != nullptr && other.data != nullptr && *data == *other.data;
521 }
522 bool operator<(const MoveOnlyDataType& other) const {
523 return data == nullptr || (other.data != nullptr && *data < *other.data);
524 }
525
526 bool operator==(int other) const { return data != nullptr && *data == other; }
527 friend bool operator==(int left, const MoveOnlyDataType& right) {
528 return right == left;
529 }
530
531 int* data = nullptr;
532 int moves = 0;
533};
