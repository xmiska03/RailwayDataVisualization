167// This code is adapted from
168// https://github.com/ray-project/ray/blob/master/src/ray/util/logging.h.
169
170// To make the logging lib pluggable with other logging libs and make
171// the implementation unawared by the user, ArrowLog is only a declaration
172// which hide the implementation into logging.cc file.
173// In logging.cc, we can choose different log libs using different macros.
174
175// This is also a null log which does not output anything.
176class ARROW_EXPORT ArrowLogBase {
177 public:
178 virtual ~ArrowLogBase() {}
179
180 virtual bool IsEnabled() const { return false; }
181
182 template <typename T>
183 ArrowLogBase& operator<<(const T& t) {
184 if (IsEnabled()) {
185 Stream() << t;
186 }
187 return *this;
188 }
189
190 protected:
191 virtual std::ostream& Stream() = 0;
192};
