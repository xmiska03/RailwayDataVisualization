134constexpr auto kScalarScratchSpaceSize = sizeof(int64_t) * 2;

136template <typename Impl>
137struct ARROW_EXPORT ArraySpanFillFromScalarScratchSpace {
138 // 16 bytes of scratch space to enable ArraySpan to be a view onto any
139 // Scalar- including binary scalars where we need to create a buffer
140 // that looks like two 32-bit or 64-bit offsets.
141 alignas(int64_t) mutable uint8_t scratch_space_[kScalarScratchSpaceSize];
142
143 private:
144 template <typename... Args>
145 explicit ArraySpanFillFromScalarScratchSpace(Args&&... args) {
146 Impl::FillScratchSpace(scratch_space_, std::forward<Args>(args)...);
147 }
148
149 ArraySpanFillFromScalarScratchSpace() = delete;
150
151 friend Impl;
152};
