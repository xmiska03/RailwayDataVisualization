106 // Convenience method - mainly for use in ARROW_LOG_* macros. This prevents unnecessary
107 // argument evaluation when log statements are stripped in certain builds
108 template <typename... Args>
109 LogMessage& Append(Args&&... args) {
110 if constexpr (sizeof...(Args) > 0) {
111 if (CheckIsEnabled()) {
112 (Stream() << ... << args);
113 }
114 }

116 }
