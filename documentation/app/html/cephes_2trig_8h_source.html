19 /* Compute sin(pi * x). */
20 template <typename T>
21 SPECFUN_HOST_DEVICE T sinpi(T x) {
22 T s = 1.0;
23
24 if (x < 0.0) {
25 x = -x;
26 s = -1.0;
27 }
28
29 T r = std::fmod(x, 2.0);
30 if (r < 0.5) {
31 return s * std::sin(M_PI * r);
32 } else if (r > 1.5) {
33 return s * std::sin(M_PI * (r - 2.0));
34 } else {
35 return -s * std::sin(M_PI * (r - 1.0));
36 }
37 }

39 /* Compute cos(pi * x) */
