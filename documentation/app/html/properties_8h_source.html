644 std::shared_ptr<WriterProperties> build() {
645 std::unordered_map<std::string, ColumnProperties> column_properties;
646 auto get = [&](const std::string& key) -> ColumnProperties& {
647 auto it = column_properties.find(key);
648 if (it == column_properties.end())
649 return column_properties[key] = default_column_properties_;
650 else
651 return it->second;
652 };
653
654 for (const auto& item : encodings_) get(item.first).set_encoding(item.second);
655 for (const auto& item : codecs_) get(item.first).set_compression(item.second);
656 for (const auto& item : codec_options_)
657 get(item.first).set_codec_options(item.second);
658 for (const auto& item : dictionary_enabled_)
659 get(item.first).set_dictionary_enabled(item.second);
660 for (const auto& item : statistics_enabled_)
661 get(item.first).set_statistics_enabled(item.second);
662 for (const auto& item : page_index_enabled_)
663 get(item.first).set_page_index_enabled(item.second);
664
665 return std::shared_ptr<WriterProperties>(new WriterProperties(
666 pool_, dictionary_pagesize_limit_, write_batch_size_, max_row_group_length_,
667 pagesize_, version_, created_by_, page_checksum_enabled_,
668 std::move(file_encryption_properties_), default_column_properties_,
669 column_properties, data_page_version_, store_decimal_as_integer_,
670 std::move(sorting_columns_)));
671 }
