37template <typename T>
38class Ndarray1DIndexer {
39 public:
40 typedef int64_t size_type;
41
42 Ndarray1DIndexer() : arr_(NULLPTR), data_(NULLPTR) {}
43
44 explicit Ndarray1DIndexer(PyArrayObject* arr) : Ndarray1DIndexer() {
45 arr_ = arr;
46 DCHECK_EQ(1, PyArray_NDIM(arr)) << "Only works with 1-dimensional arrays";
47 data_ = reinterpret_cast<uint8_t*>(PyArray_DATA(arr));
48 stride_ = PyArray_STRIDES(arr)[0];
49 }
50
51 ~Ndarray1DIndexer() = default;
52
53 int64_t size() const { return PyArray_SIZE(arr_); }
54
55 const T* data() const { return reinterpret_cast<const T*>(data_); }
56
57 bool is_strided() const { return stride_ != sizeof(T); }
58
59 T& operator[](size_type index) {
60 return *reinterpret_cast<T*>(data_ + index * stride_);
61 }
62 const T& operator[](size_type index) const {
63 return *reinterpret_cast<const T*>(data_ + index * stride_);
64 }
65
66 private:
67 PyArrayObject* arr_;
68 uint8_t* data_;
69 int64_t stride_;
70};

75static inline std::string GetNumPyTypeName(int npy_type) {
76#define TYPE_CASE(TYPE, NAME) \
77 case NPY_##TYPE: \
78 return NAME;
79
80 switch (npy_type) {
81 TYPE_CASE(BOOL, "bool")
82 TYPE_CASE(INT8, "int8")
83 TYPE_CASE(INT16, "int16")
84 TYPE_CASE(INT32, "int32")
85 TYPE_CASE(INT64, "int64")
86#if !NPY_INT32_IS_INT
87 TYPE_CASE(INT, "intc")
88#endif
89#if !NPY_INT64_IS_LONG_LONG
90 TYPE_CASE(LONGLONG, "longlong")
91#endif
92 TYPE_CASE(UINT8, "uint8")
93 TYPE_CASE(UINT16, "uint16")
94 TYPE_CASE(UINT32, "uint32")
95 TYPE_CASE(UINT64, "uint64")
96#if !NPY_INT32_IS_INT
97 TYPE_CASE(UINT, "uintc")
98#endif
99#if !NPY_INT64_IS_LONG_LONG
100 TYPE_CASE(ULONGLONG, "ulonglong")
101#endif
102 TYPE_CASE(FLOAT16, "float16")
103 TYPE_CASE(FLOAT32, "float32")
104 TYPE_CASE(FLOAT64, "float64")
105 TYPE_CASE(DATETIME, "datetime64")
106 TYPE_CASE(TIMEDELTA, "timedelta64")
107 TYPE_CASE(OBJECT, "object")
108 TYPE_CASE(VOID, "void")
109 default:
110 break;
111 }
112
113#undef TYPE_CASE
114 std::stringstream ss;
115 ss << "unrecognized type (" << npy_type << ") in GetNumPyTypeName";
116 return ss.str();
117}
