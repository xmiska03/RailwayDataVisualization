57 * DESCRIPTION:
58 *

123 /* Gamma function computed by Stirling's formula.
124 * The polynomial STIR is valid for 33 <= x <= 172.
125 */
126 SPECFUN_HOST_DEVICE inline double stirf(double x) {
127 double y, w, v;
128
129 if (x >= MAXGAM) {
130 return (std::numeric_limits<double>::infinity());
131 }
132 w = 1.0 / x;
133 w = 1.0 + w * special::cephes::polevl(w, gamma_STIR, 4);
134 y = std::exp(x);
135 if (x > MAXSTIR) { /* Avoid overflow in pow() */
136 v = std::pow(x, 0.5 * x - 0.25);
137 y = v * (v / y);
138 } else {
139 y = std::pow(x, x - 0.5) / y;
140 }
141 y = SQRTPI * y * w;
142 return (y);
143 }

146 SPECFUN_HOST_DEVICE inline double Gamma(double x) {
147 double p, q, z;
148 int i;
149 int sgngam = 1;
150
151 if (!std::isfinite(x)) {
152 return x;
153 }
154 q = std::abs(x);
155
156 if (q > 33.0) {
157 if (x < 0.0) {
158 p = std::floor(q);
159 if (p == q) {
160 gamnan:
161 set_error("Gamma", SF_ERROR_OVERFLOW, NULL);
162 return (std::numeric_limits<double>::infinity());
163 }
164 i = p;
165 if ((i & 1) == 0) {
166 sgngam = -1;
167 }
168 z = q - p;
169 if (z > 0.5) {
170 p += 1.0;
171 z = q - p;
172 }
173 z = q * sinpi(z);
174 if (z == 0.0) {
175 return (sgngam * std::numeric_limits<double>::infinity());
176 }
177 z = std::abs(z);
178 z = M_PI / (z * detail::stirf(q));
179 } else {
180 z = detail::stirf(x);
181 }
182 return (sgngam * z);
183 }
184
185 z = 1.0;
186 while (x >= 3.0) {
187 x -= 1.0;
188 z *= x;
189 }
190
191 while (x < 0.0) {
192 if (x > -1.E-9) {
193 goto small;
194 }
195 z /= x;
196 x += 1.0;
197 }
198
199 while (x < 2.0) {
200 if (x < 1.e-9) {
201 goto small;
202 }
203 z /= x;
204 x += 1.0;
205 }
206
207 if (x == 2.0) {
208 return (z);
209 }
210
211 x -= 2.0;
212 p = polevl(x, detail::gamma_P, 6);
213 q = polevl(x, detail::gamma_Q, 7);
214 return (z * p / q);
215
216 small:
217 if (x == 0.0) {
218 goto gamnan;
219 } else
220 return (z / ((1.0 + 0.5772156649015329 * x) * x));
221 }

268

275 w = lgam_sgn(q, sign);

277 if (p == q) {
