121class PARQUET_EXPORT EncodedStatistics {
122 std::string max_, min_;
123 bool is_signed_ = false;
124
125 public:
126 EncodedStatistics() = default;
127
128 const std::string& max() const { return max_; }
129 const std::string& min() const { return min_; }
130
131 int64_t null_count = 0;
132 int64_t distinct_count = 0;
133
134 bool has_min = false;
135 bool has_max = false;
136 bool has_null_count = false;
137 bool has_distinct_count = false;
138
139 // When all values in the statistics are null, it is set to true.
140 // Otherwise, at least one value is not null, or we are not sure at all.
141 // Page index requires this information to decide whether a data page
142 // is a null page or not.
143 bool all_null_value = false;
144
145 // From parquet-mr
146 // Don't write stats larger than the max size rather than truncating. The
147 // rationale is that some engines may use the minimum value in the page as
148 // the true minimum for aggregations and there is no way to mark that a
149 // value has been truncated and is a lower bound and not in the page.
150 void ApplyStatSizeLimits(size_t length) {
151 if (max_.length() > length) {
152 has_max = false;
153 max_.clear();
154 }
155 if (min_.length() > length) {
156 has_min = false;
157 min_.clear();
158 }
159 }
160
161 bool is_set() const {
162 return has_min || has_max || has_null_count || has_distinct_count;
163 }
164
165 bool is_signed() const { return is_signed_; }
166
167 void set_is_signed(bool is_signed) { is_signed_ = is_signed; }
168
169 EncodedStatistics& set_max(std::string value) {
170 max_ = std::move(value);
171 has_max = true;
172 return *this;
173 }
174
175 EncodedStatistics& set_min(std::string value) {
176 min_ = std::move(value);
177 has_min = true;
178 return *this;
179 }
180
181 EncodedStatistics& set_null_count(int64_t value) {
182 null_count = value;
183 has_null_count = true;
184 return *this;
185 }
186
187 EncodedStatistics& set_distinct_count(int64_t value) {
188 distinct_count = value;
189 has_distinct_count = true;
190 return *this;
191 }
192};
