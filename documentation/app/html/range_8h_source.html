31template <typename T>
32std::vector<T> Iota(T start, T stop) {
33 if (start > stop) {
34 return {};
35 }
36 std::vector<T> result(static_cast<size_t>(stop - start));
37 std::iota(result.begin(), result.end(), start);
38 return result;
39}

76 class RangeIter {
77 public:
78 using difference_type = int64_t;
79 using value_type = return_type;
80 using reference = const value_type&;
81 using pointer = const value_type*;
82 using iterator_category = std::forward_iterator_tag;

84#ifdef _MSC_VER
85 // msvc complains about unchecked iterators,
86 // see https://stackoverflow.com/questions/21655496/error-c4996-checked-iterators

94 RangeIter(const LazyRange<Generator>& range, int64_t index)
95 : range_(&range), index_(index) {}
96
97 const return_type operator*() const { return range_->gen_(index_); }
98
99 RangeIter operator+(difference_type length) const {
100 return RangeIter(*range_, index_ + length);

102
103 // pre-increment
104 RangeIter& operator++() {
105 ++index_;
106 return *this;
107 }
108
109 // post-increment
110 RangeIter operator++(int) {
111 auto copy = RangeIter(*this);
112 ++index_;
113 return copy;
114 }
115
116 bool operator==(const typename LazyRange<Generator>::RangeIter& other) const {
117 return this->index_ == other.index_ && this->range_ == other.range_;
118 }
119
120 bool operator!=(const typename LazyRange<Generator>::RangeIter& other) const {
121 return this->index_ != other.index_ || this->range_ != other.range_;
122 }
123
124 int64_t operator-(const typename LazyRange<Generator>::RangeIter& other) const {
125 return this->index_ - other.index_;
126 }
127
128 bool operator<(const typename LazyRange<Generator>::RangeIter& other) const {
129 return this->index_ < other.index_;
130 }
131
132 private:
133 // parent range reference
134 const LazyRange* range_;
135 // current index
136 int64_t index_;
137 };
