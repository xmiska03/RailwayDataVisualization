605 size_t nargsf, PyObject *kwnames)
606{
607#if PY_VERSION_HEX >= 0x030800B1 && !defined(PYPY_VERSION)

616 PyErr_BadInternalCall();
617 goto error;
618 }

620 PyErr_BadInternalCall();
621 goto error;
622 }

624 nposargs = (Py_ssize_t)PyVectorcall_NARGS(nargsf);
625 if (kwnames) {

627 }
