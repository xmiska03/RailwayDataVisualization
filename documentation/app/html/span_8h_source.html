31// This trait is used to check if a type R can be used to construct a span<T>.
32// Specifically, it checks if std::data(R) and std::size(R) are valid expressions
33// that may be passed to the span(T*, size_t) constructor. The reason this trait
34// is needed rather than expressing this directly in the relevant span constructor
35// is that this check requires instantiating span<T>, which would violate the
36// C++ standard if written directly in the constructor's enable_if clause
37// because span<T> is an incomplete type at that point. By defining this trait
38// instead, we add an extra level of indirection that lets us delay the
39// evaluation of the template until the first time the associated constructor
40// is actually called, at which point span<T> is a complete type.
41//
42// Note that most compilers do support the noncompliant construct, but nvcc
43// does not. See https://github.com/apache/arrow/issues/40252
44template <class T, class R, class Enable = void>
45struct ConstructibleFromDataAndSize : std::false_type {};
46
47template <class T, class R>
48struct ConstructibleFromDataAndSize<
49 span<T>, R,
50 std::void_t<decltype(span<T>{std::data(std::declval<R>()),
51 std::size(std::declval<R>())})>> : std::true_type {};

60std::span allows contiguous_iterators instead of just pointers, the enforcement

63writing code which would break when it is replaced by std::span.)");

117 constexpr bool operator==(span const& other) const {
118 if (size_ != other.size_) return false;
119
120 if constexpr (std::is_integral_v<T>) {
121 if (size_ == 0) {
122 return true; // memcmp does not handle null pointers, even if size_ == 0
123 }
124 return std::memcmp(data_, other.data_, size_bytes()) == 0;
125 } else {
126 T* ptr = data_;
127 for (T const& e : other) {
128 if (*ptr++ != e) return false;
129 }
130 return true;
131 }
132 }
133 constexpr bool operator!=(span const& other) const { return !(*this == other); }
