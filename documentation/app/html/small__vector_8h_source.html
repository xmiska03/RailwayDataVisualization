58template <typename T, size_t N, bool D = !std::is_trivially_destructible<T>::value>
59struct StaticVectorStorage : public StaticVectorStorageBase<T, N, D> {
60 using Base = StaticVectorStorageBase<T, N, D>;
61 using typename Base::storage_type;
62
63 using Base::size_;
64 using Base::static_data_;
65
66 StaticVectorStorage() noexcept = default;
67
68 constexpr storage_type* storage_ptr() { return static_data_; }
69
70 constexpr const storage_type* const_storage_ptr() const { return static_data_; }
71
72 // Adjust storage size, but don't initialize any objects
73 void bump_size(size_t addend) {
74 assert(size_ + addend <= N);
75 size_ += addend;
76 }
77
78 void ensure_capacity(size_t min_capacity) { assert(min_capacity <= N); }
79
80 // Adjust storage size, but don't destroy any objects
81 void reduce_size(size_t reduce_by) {
82 assert(reduce_by <= size_);
83 size_ -= reduce_by;
84 }
85
86 // Move objects from another storage, but don't destroy any objects currently
87 // stored in *this.
88 // You need to call destroy() first if necessary (e.g. in a
89 // move assignment operator).
90 void move_construct(StaticVectorStorage&& other) noexcept {
91 size_ = other.size_;
92 if (size_ != 0) {
93 // Use a compile-time memcpy size (N) for trivial types
94 storage_type::move_construct_several(other.static_data_, static_data_, size_, N);
95 }
96 }
97
98 constexpr size_t capacity() const { return N; }
99
100 constexpr size_t max_size() const { return N; }
101
102 void reserve(size_t n) {}
103
104 void clear() {
105 storage_type::destroy_several(static_data_, size_);
106 size_ = 0;
107 }
108};

110template <typename T, size_t N>
111struct SmallVectorStorage {
112 using storage_type = AlignedStorage<T>;
113
114 storage_type static_data_[N];
115 size_t size_ = 0;
116 storage_type* data_ = static_data_;
117 size_t dynamic_capacity_ = 0;
118
119 SmallVectorStorage() noexcept = default;
120
121 ~SmallVectorStorage() { destroy(); }
122
123 constexpr storage_type* storage_ptr() { return data_; }
124
125 constexpr const storage_type* const_storage_ptr() const { return data_; }
126
127 void bump_size(size_t addend) {
128 const size_t new_size = size_ + addend;
129 ensure_capacity(new_size);
130 size_ = new_size;
131 }
132
133 void ensure_capacity(size_t min_capacity) {
134 if (dynamic_capacity_) {
135 // Grow dynamic storage if necessary
136 if (min_capacity > dynamic_capacity_) {
137 size_t new_capacity = std::max(dynamic_capacity_ * 2, min_capacity);
138 reallocate_dynamic(new_capacity);
139 }
140 } else if (min_capacity > N) {
141 switch_to_dynamic(min_capacity);
142 }
143 }
144
145 void reduce_size(size_t reduce_by) {
146 assert(reduce_by <= size_);
147 size_ -= reduce_by;
148 }
149
150 void destroy() noexcept {
151 storage_type::destroy_several(data_, size_);
152 if (dynamic_capacity_) {
153 delete[] data_;
154 }
155 }
156
157 void move_construct(SmallVectorStorage&& other) noexcept {
158 size_ = other.size_;
159 dynamic_capacity_ = other.dynamic_capacity_;
160 if (dynamic_capacity_) {
161 data_ = other.data_;
162 other.data_ = other.static_data_;
163 other.dynamic_capacity_ = 0;
164 other.size_ = 0;
165 } else if (size_ != 0) {
166 // Use a compile-time memcpy size (N) for trivial types
167 storage_type::move_construct_several(other.static_data_, static_data_, size_, N);
168 }
169 }
170
171 constexpr size_t capacity() const { return dynamic_capacity_ ? dynamic_capacity_ : N; }
172
173 constexpr size_t max_size() const { return std::numeric_limits<size_t>::max(); }
174
175 void reserve(size_t n) {
176 if (dynamic_capacity_) {
177 if (n > dynamic_capacity_) {
178 reallocate_dynamic(n);
179 }
180 } else if (n > N) {
181 switch_to_dynamic(n);
182 }
183 }
184
185 void clear() {
186 storage_type::destroy_several(data_, size_);
187 size_ = 0;
188 }
189
190 private:
191 void switch_to_dynamic(size_t new_capacity) {
192 dynamic_capacity_ = new_capacity;
193 data_ = new storage_type[new_capacity];
194 storage_type::move_construct_several_and_destroy_source(static_data_, data_, size_);
195 }
196
197 void reallocate_dynamic(size_t new_capacity) {
198 assert(new_capacity >= size_);
199 auto new_data = new storage_type[new_capacity];
200 storage_type::move_construct_several_and_destroy_source(data_, new_data, size_);
201 delete[] data_;
202 dynamic_capacity_ = new_capacity;
203 data_ = new_data;
204 }

391 template <typename InputIt>
392 iterator insert(const_iterator insert_at, InputIt first, InputIt last) {
393 const size_t n = storage_.size_;
394 const size_t it_size = static_cast<size_t>(last - first); // XXX might be O(n)?
395 const size_t pos = static_cast<size_t>(insert_at - const_data_ptr());
396 storage_.bump_size(it_size);
397 auto* p = storage_.storage_ptr();
398 if (it_size == 0) {
399 return p[pos].get();
400 }
401 const size_t end_pos = pos + it_size;
402
403 // Move [pos; n) to [end_pos; end_pos + n - pos)
404 size_t i = n;
405 size_t j = end_pos + n - pos;
406 while (j > std::max(n, end_pos)) {
407 p[--j].move_construct(&p[--i]);
408 }
409 while (j > end_pos) {
410 p[--j].move_assign(&p[--i]);
411 }
412 assert(j == end_pos);
413 // Copy [first; last) to [pos; end_pos)
414 j = pos;
415 while (j < std::min(n, end_pos)) {
416 p[j++].assign(*first++);
417 }
418 while (j < end_pos) {
419 p[j++].construct(*first++);
420 }
421 assert(first == last);
422 return p[pos].get();
423 }

425 void resize(size_t n) {
426 const size_t old_size = storage_.size_;
427 if (n > storage_.size_) {
428 storage_.bump_size(n - old_size);
429 auto* p = storage_.storage_ptr();
430 for (size_t i = old_size; i < n; ++i) {
431 p[i].construct(T{});
432 }
433 } else {

474 template <typename InputIt>
475 void assign_by_copying(size_t n, InputIt src) {
476 const size_t old_size = storage_.size_;
477 if (n > old_size) {
478 storage_.bump_size(n - old_size);
479 auto* dest = storage_.storage_ptr();
480 for (size_t i = 0; i < old_size; ++i, ++src) {
481 dest[i].assign(*src);
482 }
483 for (size_t i = old_size; i < n; ++i, ++src) {
484 dest[i].construct(*src);
485 }
486 } else {
487 auto* dest = storage_.storage_ptr();
488 for (size_t i = 0; i < n; ++i, ++src) {
489 dest[i].assign(*src);
490 }
491 for (size_t i = n; i < old_size; ++i) {
492 dest[i].destroy();
493 }
494 storage_.reduce_size(old_size - n);
495 }
496 }
