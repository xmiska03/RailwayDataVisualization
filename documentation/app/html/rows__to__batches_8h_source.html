32// Default identity function row accessor. Used to for the common case where the value
33// of each row iterated over is it's self also directly iterable.

113template <class Range, class DataPointConvertor,
114 class RowAccessor = decltype(detail::MakeDefaultRowAccessor())>
115[[nodiscard]] typename std::enable_if_t<detail::is_range<Range>::value,
116 Result<std::shared_ptr<RecordBatchReader>>>
117/* Result<std::shared_ptr<RecordBatchReader>>> */ RowsToBatches(
118 const std::shared_ptr<Schema>& schema, const Range& rows,
119 DataPointConvertor&& data_point_convertor,
120 RowAccessor&& row_accessor = detail::MakeDefaultRowAccessor(),
121 MemoryPool* pool = default_memory_pool(), const std::size_t batch_size = 1024) {
122 auto make_next_batch =
123 [pool = pool, batch_size = batch_size, rows_ittr = std::begin(rows),
124 rows_ittr_end = std::end(rows), schema = schema,
125 row_accessor = std::forward<RowAccessor>(row_accessor),
126 data_point_convertor = std::forward<DataPointConvertor>(
127 data_point_convertor)]() mutable -> Result<std::shared_ptr<RecordBatch>> {
128 if (rows_ittr == rows_ittr_end) return NULLPTR;
129
130 ARROW_ASSIGN_OR_RAISE(auto record_batch_builder,
131 RecordBatchBuilder::Make(schema, pool, batch_size));
132
133 for (size_t i = 0; i < batch_size && (rows_ittr != rows_ittr_end);
134 i++, std::advance(rows_ittr, 1)) {
135 int col_index = 0;
136 ARROW_ASSIGN_OR_RAISE(const auto row, row_accessor(*rows_ittr));
137
138 // If the accessor returns a `std::reference_wrapper` unwrap if
139 const auto& row_unwrapped = [&]() {
140 if constexpr (detail::is_range<decltype(row)>::value)
141 return row;
142 else
143 return row.get();
144 }();
145
146 for (auto& data_point : row_unwrapped) {
147 ArrayBuilder* array_builder = record_batch_builder->GetField(col_index);
148 ARROW_RETURN_IF(array_builder == NULLPTR,
149 Status::Invalid("array_builder == NULLPTR"));
150
151 ARROW_RETURN_NOT_OK(data_point_convertor(*array_builder, data_point));
152 col_index++;
153 }
154 }
155
156 ARROW_ASSIGN_OR_RAISE(auto result, record_batch_builder->Flush());
157 return result;
158 };
159 return RecordBatchReader::MakeFromIterator(MakeFunctionIterator(make_next_batch),
160 schema);
161}
