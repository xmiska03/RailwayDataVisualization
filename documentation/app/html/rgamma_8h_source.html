15 * DESCRIPTION:
16 *
17 * Returns one divided by the Gamma function of the argument.
18 *
19 * The function is approximated by a Chebyshev expansion in

34 * Relative error:

74 SPECFUN_HOST_DEVICE double rgamma(double x) {
75 double w, y, z;
76 int sign;
77
78 if (x > 34.84425627277176174) {
79 return std::exp(-lgam(x));
80 }
81 if (x < -34.034) {
82 w = -x;
83 z = sinpi(w);
84 if (z == 0.0) {
85 return 0.0;
86 }
87 if (z < 0.0) {
88 sign = 1;
89 z = -z;
90 } else {
91 sign = -1;
92 }
93
94 y = std::log(w * z) - std::log(M_PI) + lgam(w);
95 if (y < -detail::MAXLOG) {
96 set_error("rgamma", SF_ERROR_UNDERFLOW, NULL);
97 return (sign * 0.0);
98 }
99 if (y > detail::MAXLOG) {
100 set_error("rgamma", SF_ERROR_OVERFLOW, NULL);
101 return (sign * std::numeric_limits<double>::infinity());
102 }
103 return (sign * std::exp(y));
104 }
105 z = 1.0;
106 w = x;
107
108 while (w > 1.0) { /* Downward recurrence */
109 w -= 1.0;
110 z *= w;
111 }
112 while (w < 0.0) { /* Upward recurrence */
113 z /= w;
114 w += 1.0;
115 }
116 if (w == 0.0) /* Nonpositive integer */
117 return (0.0);
118 if (w == 1.0) /* Other integer */
119 return (1.0 / z);
120
121 y = w * (1.0 + chbevl(4.0 * w - 2.0, detail::rgamma_R, 16)) / z;
122 return (y);
123 }
