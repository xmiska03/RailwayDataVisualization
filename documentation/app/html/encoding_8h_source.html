165// Base class for value encoders. Since encoders may or not have state (e.g.,
166// dictionary encoding) we use a class instance to maintain any state.
167//
168// Encode interfaces are internal, subject to change without deprecation.
169template <typename DType>
170class TypedEncoder : virtual public Encoder {
171 public:
172 using T = typename DType::c_type;
173
174 using Encoder::Put;
175
176 virtual void Put(const T* src, int num_values) = 0;
177
178 virtual void Put(const std::vector<T>& src, int num_values = -1);
179
180 virtual void PutSpaced(const T* src, int num_values, const uint8_t* valid_bits,
181 int64_t valid_bits_offset) = 0;
182};

184template <typename DType>
185void TypedEncoder<DType>::Put(const std::vector<T>& src, int num_values) {
186 if (num_values == -1) {
187 num_values = static_cast<int>(src.size());
188 }
189 Put(src.data(), num_values);
190}

234// ----------------------------------------------------------------------
235// Value decoding

237class Decoder {
238 public:
239 virtual ~Decoder() = default;
240
241 // Sets the data for a new page. This will be called multiple times on the same
242 // decoder and should reset all internal state.
243 //
244 // `num_values` comes from the data page header, and may be greater than the number of
245 // physical values in the data buffer if there are some omitted (null) values.
246 // `len`, on the other hand, is the size in bytes of the data buffer and
247 // directly relates to the number of physical values.
248 virtual void SetData(int num_values, const uint8_t* data, int len) = 0;
249
250 // Returns the number of values left (for the last call to SetData()). This is
251 // the number of values left in this page.
252 virtual int values_left() const = 0;
253 virtual Encoding::type encoding() const = 0;
254};
