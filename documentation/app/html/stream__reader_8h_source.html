245 template <typename ReaderType, typename ReadType = typename ReaderType::T, typename T>
246 void ReadOptional(optional<T>* v) {
247 const auto& node = nodes_[column_index_];
248 auto reader = static_cast<ReaderType*>(column_readers_[column_index_++].get());
249 int16_t def_level;
250 int16_t rep_level;
251 ReadType tmp;
252 int64_t values_read;
253
254 reader->ReadBatch(kBatchSizeOne, &def_level, &rep_level, &tmp, &values_read);
255
256 if (values_read == 1) {
257 *v = T(tmp);
258 } else if ((values_read == 0) && (def_level == 0)) {
259 v->reset();
260 } else {
261 ThrowReadFailedException(node);
262 }
263 }
