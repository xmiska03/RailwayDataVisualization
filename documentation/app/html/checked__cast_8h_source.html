28inline OutputType checked_cast(InputType&& value) {

30 typename std::remove_reference<InputType>::type>::type>::value,
31 "checked_cast input type must be a class");
32 static_assert(std::is_class<typename std::remove_pointer<
33 typename std::remove_reference<OutputType>::type>::type>::value,
34 "checked_cast output type must be a class");
35#ifdef NDEBUG
36 return static_cast<OutputType>(value);
