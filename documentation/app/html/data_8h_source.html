220 return bit_util::GetBit(buffers[0]->data(), i + offset);
221 }

224 return !internal::IsNullSparseUnion(*this, i);
225 }
226 if (type == Type::DENSE_UNION) {
227 return !internal::IsNullDenseUnion(*this, i);
228 }
229 if (type == Type::RUN_END_ENCODED) {
230 return !internal::IsNullRunEndEncoded(*this, i);
231 }
232 return null_count.load() != length;
233 }
