114 // This is an RAII class for keeping track of in-flight file IO. Useful for getting
115 // an estimate of memory use, and how much memory we expect to be freed soon.
116 // Returned by ReportTempFileIO.
117 struct [[nodiscard]] TempFileIOMark {
118 QueryContext* ctx_;
119 size_t bytes_;
120
121 TempFileIOMark(QueryContext* ctx, size_t bytes) : ctx_(ctx), bytes_(bytes) {
122 ctx_->in_flight_bytes_to_disk_.fetch_add(bytes_, std::memory_order_acquire);
123 }
124
125 ARROW_DISALLOW_COPY_AND_ASSIGN(TempFileIOMark);
126
127 ~TempFileIOMark() {
128 ctx_->in_flight_bytes_to_disk_.fetch_sub(bytes_, std::memory_order_release);
129 }
130 };
