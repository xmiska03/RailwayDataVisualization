514inline Result<int32_t> MaxDecimalDigitsForInteger(Type::type type_id) {
515 switch (type_id) {
516 case Type::INT8:
517 case Type::UINT8:
518 return 3;
519 case Type::INT16:
520 case Type::UINT16:
521 return 5;
522 case Type::INT32:
523 case Type::UINT32:
524 return 10;
525 case Type::INT64:
526 return 19;
527 case Type::UINT64:
528 return 20;
529 default:
530 break;
531 }
532 return Status::Invalid("Not an integer type: ", type_id);
533}
