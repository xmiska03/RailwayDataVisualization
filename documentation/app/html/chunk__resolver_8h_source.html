167 template <typename IndexType>
168 [[nodiscard]] bool ResolveMany(int64_t n_indices, const IndexType* logical_index_vec,
169 TypedChunkLocation<IndexType>* out_chunk_location_vec,
170 IndexType chunk_hint = 0) const {
171 if constexpr (sizeof(IndexType) < sizeof(uint32_t)) {
172 // The max value returned by Bisect is `offsets.size() - 1` (= chunks.size()).
173 constexpr int64_t kMaxIndexTypeValue = std::numeric_limits<IndexType>::max();
174 // A ChunkedArray with enough empty chunks can make the index of a chunk
175 // exceed the logical index and thus the maximum value of IndexType.
176 const bool chunk_index_fits_on_type = num_chunks() <= kMaxIndexTypeValue;
177 if (ARROW_PREDICT_FALSE(!chunk_index_fits_on_type)) {
178 return false;
179 }
180 // Since an index-in-chunk cannot possibly exceed the logical index being
181 // queried, we don't have to worry about these values not fitting on IndexType.
182 }
183 if constexpr (std::is_signed_v<IndexType>) {
184 // We interpret signed integers as unsigned and avoid having to generate double
185 // the amount of binary code to handle each integer width.
186 //
187 // Negative logical indices can become large values when cast to unsigned, and
188 // they are gracefully handled by ResolveManyImpl, but both the chunk index
189 // and the index in chunk values will be undefined in these cases. This
190 // happend because int8_t(-1) == uint8_t(255) and 255 could be a valid
191 // logical index in the chunked array.
192 using U = std::make_unsigned_t<IndexType>;
193 ResolveManyImpl(n_indices, reinterpret_cast<const U*>(logical_index_vec),
194 reinterpret_cast<TypedChunkLocation<U>*>(out_chunk_location_vec),
195 static_cast<int32_t>(chunk_hint));
196 } else {
197 static_assert(std::is_unsigned_v<IndexType>);
198 ResolveManyImpl(n_indices, logical_index_vec, out_chunk_location_vec,
199 static_cast<int32_t>(chunk_hint));
200 }
201 return true;
202 }

205 template <bool StoreCachedChunk>
206 inline int64_t ResolveChunkIndex(int64_t index, int32_t cached_chunk) const {
207 // It is common for algorithms sequentially processing arrays to make consecutive
208 // accesses at a relatively small distance from each other, hence often falling in the
209 // same chunk.
210 //
211 // This is guaranteed when merging (assuming each side of the merge uses its
212 // own resolver), and is the most common case in recursive invocations of
213 // partitioning.
214 const auto num_offsets = static_cast<uint32_t>(offsets_.size());
215 const int64_t* offsets = offsets_.data();
216 if (ARROW_PREDICT_TRUE(index >= offsets[cached_chunk]) &&
217 (static_cast<uint32_t>(cached_chunk + 1) == num_offsets ||
218 index < offsets[cached_chunk + 1])) {
219 return cached_chunk;
220 }
221 // lo < hi is guaranteed by `num_offsets = chunks.size() + 1`
222 const auto chunk_index = Bisect(index, offsets, /*lo=*/0, /*hi=*/num_offsets);
223 if constexpr (StoreCachedChunk) {
224 assert(static_cast<uint32_t>(chunk_index) < static_cast<uint32_t>(offsets_.size()));
225 cached_chunk_.store(chunk_index, std::memory_order_relaxed);
226 }
227 return chunk_index;
228 }

259 static inline int32_t Bisect(uint64_t index, const uint64_t* offsets, uint32_t lo,
260 uint32_t hi) {
261 // Similar to std::upper_bound(), but slightly different as our offsets
262 // array always starts with 0.
263 auto n = hi - lo;
264 // First iteration does not need to check for n > 1
265 // (lo < hi is guaranteed by the precondition).
266 assert(n > 1 && "lo < hi is a precondition of Bisect");
267 do {
268 const uint32_t m = n >> 1;
269 const uint32_t mid = lo + m;
270 if (index >= offsets[mid]) {
271 lo = mid;
272 n -= m;
273 } else {
274 n = m;
275 }
276 } while (n > 1);
277 return lo;
278 }
