46template <typename T>
47inline T* MakeNonNull(T* maybe_null = NULLPTR) {
48 if (ARROW_PREDICT_TRUE(maybe_null != NULLPTR)) {
49 return maybe_null;
50 }

52 return const_cast<T*>(reinterpret_cast<const T*>(&internal::kNonNullFiller));
