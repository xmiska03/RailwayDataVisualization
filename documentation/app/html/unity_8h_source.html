53 SPECFUN_HOST_DEVICE inline double log1p(double x) {
54 double z;
55
56 z = 1.0 + x;
57 if ((z < M_SQRT1_2) || (z > M_SQRT2))
58 return (std::log(z));
59 z = x * x;
60 z = -0.5 * z + x * (z * polevl(x, detail::unity_LP, 6) / p1evl(x, detail::unity_LQ, 6));
61 return (x + z);

109 SPECFUN_HOST_DEVICE inline double expm1(double x) {
110 double r, xx;
111
112 if (!std::isfinite(x)) {
113 if (std::isnan(x)) {
114 return x;
115 } else if (x > 0) {
116 return x;
117 } else {
118 return -1.0;
119 }
120 }
121 if ((x < -0.5) || (x > 0.5))
122 return (std::exp(x) - 1.0);
123 xx = x * x;
124 r = x * polevl(xx, detail::unity_EP, 2);
125 r = r / (polevl(xx, detail::unity_EQ, 3) - r);
126 return (r + r);
