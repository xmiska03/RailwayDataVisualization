56 // this function is intensively called and performance critical
57 // call it only if you are sure no NAN exists in input data
58 void Add(double value) {
59 ARROW_DCHECK(!std::isnan(value)) << "cannot add NAN";
60 if (ARROW_PREDICT_FALSE(input_.size() == input_.capacity())) {
61 MergeInput();
62 }
63 input_.push_back(value);
64 }
