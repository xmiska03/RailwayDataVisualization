85inline const uint8_t* str2bytes(const std::string& str) {
86 if (str.empty()) return NULLPTR;
87
88 char* cbytes = const_cast<char*>(str.c_str());
89 return reinterpret_cast<const uint8_t*>(cbytes);
90}

92inline ::arrow::util::span<const uint8_t> str2span(const std::string& str) {
93 if (str.empty()) {
94 return {};
95 }
96
97 return {reinterpret_cast<const uint8_t*>(str.data()), str.size()};
