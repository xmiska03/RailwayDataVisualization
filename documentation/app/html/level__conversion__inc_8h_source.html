236inline uint64_t ExtractBitsSoftware(uint64_t bitmap, uint64_t select_bitmap) {
237 // A software emulation of _pext_u64
238
239 // These checks should be inline and are likely to be common cases.
240 if (select_bitmap == ~uint64_t{0}) {
241 return bitmap;
242 } else if (select_bitmap == 0) {
243 return 0;
244 }
245
246 // Fallback to lookup table method
247 uint64_t bit_value = 0;
248 int bit_len = 0;
249 constexpr uint8_t kLookupMask = (1U << kLookupBits) - 1;
250 while (select_bitmap != 0) {
251 const auto mask_len = ARROW_POPCOUNT32(select_bitmap & kLookupMask);
252 const uint64_t value = kPextTable[select_bitmap & kLookupMask][bitmap & kLookupMask];
253 bit_value |= (value << bit_len);
254 bit_len += mask_len;
255 bitmap >>= kLookupBits;
256 select_bitmap >>= kLookupBits;
257 }
258 return bit_value;
259}

295template <bool has_repeated_parent>
296int64_t DefLevelsBatchToBitmap(const int16_t* def_levels, const int64_t batch_size,
297 int64_t upper_bound_remaining, LevelInfo level_info,
298 ::arrow::internal::FirstTimeBitmapWriter* writer) {
299 ARROW_DCHECK_LE(batch_size, kExtractBitsSize);
300
301 // Greater than level_info.def_level - 1 implies >= the def_level
302 auto defined_bitmap = static_cast<extract_bitmap_t>(
303 internal::GreaterThanBitmap(def_levels, batch_size, level_info.def_level - 1));
304
305 if (has_repeated_parent) {
306 // Greater than level_info.repeated_ancestor_def_level - 1 implies >= the
307 // repeated_ancestor_def_level
308 auto present_bitmap = static_cast<extract_bitmap_t>(internal::GreaterThanBitmap(
309 def_levels, batch_size, level_info.repeated_ancestor_def_level - 1));
310 auto selected_bits = ExtractBits(defined_bitmap, present_bitmap);
311 int64_t selected_count = ::arrow::bit_util::PopCount(present_bitmap);
312 if (ARROW_PREDICT_FALSE(selected_count > upper_bound_remaining)) {
313 throw ParquetException("Values read exceeded upper bound");
314 }
315 writer->AppendWord(selected_bits, selected_count);
316 return ::arrow::bit_util::PopCount(selected_bits);
317 } else {
318 if (ARROW_PREDICT_FALSE(batch_size > upper_bound_remaining)) {
319 std::stringstream ss;
320 ss << "Values read exceeded upper bound";
321 throw ParquetException(ss.str());
322 }
323
324 writer->AppendWord(defined_bitmap, batch_size);
325 return ::arrow::bit_util::PopCount(defined_bitmap);
326 }
327}

329template <bool has_repeated_parent>
330void DefLevelsToBitmapSimd(const int16_t* def_levels, int64_t num_def_levels,
331 LevelInfo level_info, ValidityBitmapInputOutput* output) {
332 ::arrow::internal::FirstTimeBitmapWriter writer(
333 output->valid_bits,
334 /*start_offset=*/output->valid_bits_offset,
335 /*length=*/output->values_read_upper_bound);
336 int64_t set_count = 0;
337 output->values_read = 0;
338 int64_t values_read_remaining = output->values_read_upper_bound;
339 while (num_def_levels > kExtractBitsSize) {
340 set_count += DefLevelsBatchToBitmap<has_repeated_parent>(
341 def_levels, kExtractBitsSize, values_read_remaining, level_info, &writer);
342 def_levels += kExtractBitsSize;
343 num_def_levels -= kExtractBitsSize;
344 values_read_remaining = output->values_read_upper_bound - writer.position();
345 }
346 set_count += DefLevelsBatchToBitmap<has_repeated_parent>(
347 def_levels, num_def_levels, values_read_remaining, level_info, &writer);
348
349 output->values_read = writer.position();
350 output->null_count += output->values_read - set_count;
351 writer.Finish();
352}
