133 /* Compute
134 *
135 * x^a * exp(-x) / gamma(a)
136 *
137 * corrected from (15) and (16) in [2] by replacing exp(x - a) with
138 * exp(a - x).
139 */
140 SPECFUN_HOST_DEVICE inline double igam_fac(double a, double x) {
141 double ax, fac, res, num;
142
143 if (std::abs(a - x) > 0.4 * std::abs(a)) {
144 ax = a * std::log(x) - x - special::cephes::lgam(a);
145 if (ax < -MAXLOG) {
146 set_error("igam", SF_ERROR_UNDERFLOW, NULL);
147 return 0.0;
148 }
149 return std::exp(ax);
150 }
151
152 fac = a + special::cephes::lanczos_g - 0.5;
153 res = std::sqrt(fac / std::exp(1)) / special::cephes::lanczos_sum_expg_scaled(a);
154
155 if ((a < 200) && (x < 200)) {
156 res *= std::exp(a - x) * std::pow(x / fac, a);
157 } else {
158 num = x - a - special::cephes::lanczos_g + 0.5;
159 res *= std::exp(a * special::cephes::log1pmx(num / fac) + x * (0.5 - special::cephes::lanczos_g) / fac);
160 }
161
162 return res;
163 }

165 /* Compute igamc using DLMF 8.9.2. */
166 SPECFUN_HOST_DEVICE inline double igamc_continued_fraction(double a, double x) {
167 int i;
168 double ans, ax, c, yc, r, t, y, z;
169 double pk, pkm1, pkm2, qk, qkm1, qkm2;
170
171 ax = igam_fac(a, x);
172 if (ax == 0.0) {
173 return 0.0;
174 }
175
176 /* continued fraction */
177 y = 1.0 - a;
178 z = x + y + 1.0;
179 c = 0.0;
180 pkm2 = 1.0;
181 qkm2 = x;
182 pkm1 = x + 1.0;
183 qkm1 = z * x;
184 ans = pkm1 / qkm1;
185
186 for (i = 0; i < igam_MAXITER; i++) {
187 c += 1.0;
188 y += 1.0;
189 z += 2.0;
190 yc = y * c;
191 pk = pkm1 * z - pkm2 * yc;
192 qk = qkm1 * z - qkm2 * yc;
193 if (qk != 0) {
194 r = pk / qk;
195 t = std::abs((ans - r) / r);
196 ans = r;
197 } else
198 t = 1.0;
199 pkm2 = pkm1;
200 pkm1 = pk;
201 qkm2 = qkm1;
202 qkm1 = qk;
203 if (std::abs(pk) > igam_big) {
204 pkm2 *= igam_biginv;
205 pkm1 *= igam_biginv;
206 qkm2 *= igam_biginv;
207 qkm1 *= igam_biginv;
208 }
209 if (t <= MACHEP) {
210 break;
211 }
212 }
213
214 return (ans * ax);
215 }

217 /* Compute igam using DLMF 8.11.4. */
218 SPECFUN_HOST_DEVICE inline double igam_series(double a, double x) {
219 int i;
220 double ans, ax, c, r;
221
222 ax = igam_fac(a, x);
223 if (ax == 0.0) {
224 return 0.0;
225 }
226
227 /* power series */
228 r = a;
229 c = 1.0;
230 ans = 1.0;
231
232 for (i = 0; i < igam_MAXITER; i++) {
233 r += 1.0;
234 c *= x / r;
235 ans += c;
236 if (c <= MACHEP * ans) {
237 break;
238 }
239 }
240
241 return (ans * ax / a);
242 }

244 /* Compute igamc using DLMF 8.7.3. This is related to the series in
245 * igam_series but extra care is taken to avoid cancellation.
246 */
247 SPECFUN_HOST_DEVICE inline double igamc_series(double a, double x) {
248 int n;
249 double fac = 1;
250 double sum = 0;
251 double term, logx;
252
253 for (n = 1; n < igam_MAXITER; n++) {
254 fac *= -x / n;
255 term = fac / (a + n);
256 sum += term;
257 if (std::abs(term) <= MACHEP * std::abs(sum)) {
258 break;
259 }
260 }
261
262 logx = std::log(x);
263 term = -special::cephes::expm1(a * logx - special::cephes::lgam1p(a));
264 return term - std::exp(a * logx - special::cephes::lgam(a)) * sum;
265 }

268 SPECFUN_HOST_DEVICE inline double asymptotic_series(double a, double x, int func) {
269 int k, n, sgn;
270 int maxpow = 0;
271 double lambda = x / a;
272 double sigma = (x - a) / a;
273 double eta, res, ck, ckterm, term, absterm;
274 double absoldterm = std::numeric_limits<double>::infinity();
275 double etapow[detail::igam_asymp_coeff_N] = {1};
276 double sum = 0;
277 double afac = 1;
278
279 if (func == detail::IGAM) {
280 sgn = -1;
281 } else {
282 sgn = 1;
283 }
284
285 if (lambda > 1) {
286 eta = std::sqrt(-2 * special::cephes::log1pmx(sigma));
287 } else if (lambda < 1) {
288 eta = -std::sqrt(-2 * special::cephes::log1pmx(sigma));
289 } else {
290 eta = 0;
291 }
292 res = 0.5 * special::cephes::erfc(sgn * eta * std::sqrt(a / 2));
293
294 for (k = 0; k < igam_asymp_coeff_K; k++) {
295 ck = igam_asymp_coeff_d[k][0];
296 for (n = 1; n < igam_asymp_coeff_N; n++) {
297 if (n > maxpow) {
298 etapow[n] = eta * etapow[n - 1];
299 maxpow += 1;
300 }
301 ckterm = igam_asymp_coeff_d[k][n] * etapow[n];
302 ck += ckterm;
303 if (std::abs(ckterm) < MACHEP * std::abs(ck)) {
304 break;
305 }
306 }
307 term = ck * afac;
308 absterm = std::abs(term);
309 if (absterm > absoldterm) {
310 break;
311 }
312 sum += term;
313 if (absterm < MACHEP * std::abs(sum)) {
314 break;
315 }
316 absoldterm = absterm;
317 afac /= a;
318 }
319 res += sgn * std::exp(-0.5 * a * eta * eta) * sum / std::sqrt(2 * M_PI * a);
320
321 return res;
322 }

328 SPECFUN_HOST_DEVICE inline double igam(double a, double x) {
329 double absxma_a;
330
331 if (x < 0 || a < 0) {
332 set_error("gammainc", SF_ERROR_DOMAIN, NULL);
333 return std::numeric_limits<double>::quiet_NaN();
334 } else if (a == 0) {
335 if (x > 0) {
336 return 1;
337 } else {
338 return std::numeric_limits<double>::quiet_NaN();
339 }
340 } else if (x == 0) {
341 /* Zero integration limit */
342 return 0;
343 } else if (std::isinf(a)) {
344 if (std::isinf(x)) {
345 return std::numeric_limits<double>::quiet_NaN();
346 }
347 return 0;
348 } else if (std::isinf(x)) {
349 return 1;
350 }
351
352 /* Asymptotic regime where a ~ x; see [2]. */
353 absxma_a = std::abs(x - a) / a;
354 if ((a > detail::igam_SMALL) && (a < detail::igam_LARGE) && (absxma_a < detail::igam_SMALLRATIO)) {
355 return detail::asymptotic_series(a, x, detail::IGAM);
356 } else if ((a > detail::igam_LARGE) && (absxma_a < detail::igam_LARGERATIO / std::sqrt(a))) {
357 return detail::asymptotic_series(a, x, detail::IGAM);
358 }
359
360 if ((x > 1.0) && (x > a)) {
361 return (1.0 - igamc(a, x));
362 }
363
364 return detail::igam_series(a, x);
365 }

367 SPECFUN_HOST_DEVICE double igamc(double a, double x) {
368 double absxma_a;
369
370 if (x < 0 || a < 0) {
371 set_error("gammaincc", SF_ERROR_DOMAIN, NULL);
372 return std::numeric_limits<double>::quiet_NaN();
373 } else if (a == 0) {
374 if (x > 0) {
375 return 0;
376 } else {
377 return std::numeric_limits<double>::quiet_NaN();
378 }
379 } else if (x == 0) {
380 return 1;
381 } else if (std::isinf(a)) {
382 if (std::isinf(x)) {
383 return std::numeric_limits<double>::quiet_NaN();
384 }
385 return 1;
386 } else if (std::isinf(x)) {
387 return 0;
388 }
389
390 /* Asymptotic regime where a ~ x; see [2]. */
391 absxma_a = std::abs(x - a) / a;
392 if ((a > detail::igam_SMALL) && (a < detail::igam_LARGE) && (absxma_a < detail::igam_SMALLRATIO)) {
393 return detail::asymptotic_series(a, x, detail::IGAMC);
394 } else if ((a > detail::igam_LARGE) && (absxma_a < detail::igam_LARGERATIO / std::sqrt(a))) {
395 return detail::asymptotic_series(a, x, detail::IGAMC);
396 }
397
398 /* Everywhere else; see [2]. */
399 if (x > 1.1) {
400 if (x < a) {
401 return 1.0 - detail::igam_series(a, x);
402 } else {
403 return detail::igamc_continued_fraction(a, x);
404 }
405 } else if (x <= 0.5) {
406 if (-0.4 / std::log(x) < a) {
407 return 1.0 - detail::igam_series(a, x);
408 } else {
409 return detail::igamc_series(a, x);
410 }
411 } else {
412 if (x * 1.1 < a) {
413 return 1.0 - detail::igam_series(a, x);
414 } else {
415 return detail::igamc_series(a, x);
416 }
417 }
418 }
