125 SPECFUN_HOST_DEVICE inline double i1(double x) {
126 double y, z;
127
128 z = std::abs(x);
129 if (z <= 8.0) {
130 y = (z / 2.0) - 2.0;
131 z = chbevl(y, detail::i1_A, 29) * z * std::exp(z);
132 } else {
133 z = std::exp(z) * chbevl(32.0 / z - 2.0, detail::i1_B, 25) / std::sqrt(z);
134 }
135 if (x < 0.0)
136 z = -z;
137 return (z);
138 }
