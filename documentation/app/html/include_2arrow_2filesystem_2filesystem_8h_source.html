360struct FileSystemFactory {
361 std::function<Result<std::shared_ptr<FileSystem>>(
362 const Uri& uri, const io::IOContext& io_context, std::string* out_path)>
363 function;
364 std::string_view file;
365 int line;
366
367 bool operator==(const FileSystemFactory& other) const {
368 // In the case where libarrow is linked statically both to the executable and to a
369 // dynamically loaded filesystem implementation library, the library contains a
370 // duplicate definition of the registry and duplicate definitions of any
371 // FileSystemRegistrars which are statically linked to libarrow. When retrieving
372 // factories from the filesystem implementation library, we use the file and line
373 // of the registrar's definition to determine equivalence of the duplicate factories.
374 return file == other.file && line == other.line;
375 }
