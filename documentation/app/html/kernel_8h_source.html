445struct MemAllocation {
446 enum type {
447 // For data types that support pre-allocation (i.e. fixed-width), the
448 // kernel expects to be provided a pre-allocated data buffer to write
449 // into. Non-fixed-width types must always allocate their own data
450 // buffers. The allocation made for the same length as the execution batch,
451 // so vector kernels yielding differently sized output should not use this.
452 //
453 // It is valid for the data to not be preallocated but the validity bitmap
454 // is (or is computed using the intersection/bitwise-and method).
455 //
456 // For variable-size output types like BinaryType or StringType, or for
457 // nested types, this option has no effect.
458 PREALLOCATE,
459
460 // The kernel is responsible for allocating its own data buffer for
461 // fixed-width type outputs.
462 NO_PREALLOCATE
463 };
464};
