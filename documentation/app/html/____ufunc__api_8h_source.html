252static inline int
253_import_umath(void)
254{
255 PyObject *numpy = PyImport_ImportModule("numpy._core._multiarray_umath");
256 if (numpy == NULL && PyErr_ExceptionMatches(PyExc_ModuleNotFoundError)) {
257 PyErr_Clear();
258 numpy = PyImport_ImportModule("numpy.core._multiarray_umath");
259 }
260
261 if (numpy == NULL) {
262 PyErr_SetString(PyExc_ImportError,
263 "_multiarray_umath failed to import");
264 return -1;
265 }
266
267 PyObject *c_api = PyObject_GetAttrString(numpy, "_UFUNC_API");
268 Py_DECREF(numpy);
269 if (c_api == NULL) {
270 PyErr_SetString(PyExc_AttributeError, "_UFUNC_API not found");
271 return -1;
272 }
273
274 if (!PyCapsule_CheckExact(c_api)) {
275 PyErr_SetString(PyExc_RuntimeError, "_UFUNC_API is not PyCapsule object");
276 Py_DECREF(c_api);
277 return -1;
278 }
279 PyUFunc_API = (void **)PyCapsule_GetPointer(c_api, NULL);
280 Py_DECREF(c_api);
281 if (PyUFunc_API == NULL) {
282 PyErr_SetString(PyExc_RuntimeError, "_UFUNC_API is NULL pointer");
283 return -1;
284 }
285 return 0;
286}
