67template <typename RunEndCType>
68int64_t FindPhysicalIndex(const RunEndCType* run_ends, int64_t run_ends_size, int64_t i,
69 int64_t absolute_offset) {
70 assert(absolute_offset + i >= 0);
71 auto it = std::upper_bound(run_ends, run_ends + run_ends_size, absolute_offset + i);
72 int64_t result = std::distance(run_ends, it);
73 assert(result <= run_ends_size);
74 return result;
75}

82template <typename RunEndCType>
83std::pair<int64_t, int64_t> FindPhysicalRange(const RunEndCType* run_ends,
84 int64_t run_ends_size, int64_t length,
85 int64_t offset) {
86 const int64_t physical_offset =
87 FindPhysicalIndex<RunEndCType>(run_ends, run_ends_size, 0, offset);
88 // The physical length is calculated by finding the offset of the last element
89 // and adding 1 to it, so first we ensure there is at least one element.
90 if (length == 0) {
91 return {physical_offset, 0};
92 }
93 const int64_t physical_index_of_last = FindPhysicalIndex<RunEndCType>(
94 run_ends + physical_offset, run_ends_size - physical_offset, length - 1, offset);
95
96 assert(physical_index_of_last < run_ends_size - physical_offset);
97 return {physical_offset, physical_index_of_last + 1};
98}

103template <typename RunEndCType>
104int64_t FindPhysicalLength(const RunEndCType* run_ends, int64_t run_ends_size,
105 int64_t length, int64_t offset) {
106 auto [_, physical_length] =
107 FindPhysicalRange<RunEndCType>(run_ends, run_ends_size, length, offset);
108 // GH-37107: This is a workaround for GCC 7. GCC 7 doesn't ignore
109 // variables in structured binding automatically from unused
110 // variables when one of these variables are used.
111 // See also: https://gcc.gnu.org/bugzilla/show_bug.cgi?id=81767
112 ARROW_UNUSED(_);
113 return physical_length;
114}

174template <typename RunEndCType>
175struct PhysicalIndexFinder {
176 const ArraySpan array_span;
177 const RunEndCType* run_ends;
178 int64_t last_physical_index = 0;
179
180 explicit PhysicalIndexFinder(const ArrayData& data)
181 : array_span(data),
182 run_ends(RunEndsArray(array_span).template GetValues<RunEndCType>(1)) {
183 assert(CTypeTraits<RunEndCType>::ArrowType::type_id ==
184 ::arrow::internal::checked_cast<const RunEndEncodedType&>(*data.type)

186 ->id());

188
194 int64_t FindPhysicalIndex(int64_t i) {
195 if constexpr (std::is_same_v<RunEndCType, int16_t>) {
196 return FindPhysicalIndexImpl16(*this, i);
197 } else if constexpr (std::is_same_v<RunEndCType, int32_t>) {
198 return FindPhysicalIndexImpl32(*this, i);
199 } else {
200 static_assert(std::is_same_v<RunEndCType, int64_t>, "Unsupported RunEndCType.");
201 return FindPhysicalIndexImpl64(*this, i);
202 }
203 }

373 Iterator iterator(int64_t logical_pos) const {
374 if (logical_pos < length()) {
375 return iterator(logical_pos, PhysicalIndex(logical_pos));
376 }
377 // If logical_pos is above the valid range, use length() as the logical
378 // position and calculate the physical address right after the last valid
379 // physical position. Which is the physical index of the last logical
380 // position, plus 1.
381 return (length() == 0) ? iterator(0, PhysicalIndex(0))
382 : iterator(length(), PhysicalIndex(length() - 1) + 1);
383 }

409 Iterator end() const {
410 return iterator(length(),
411 (length() == 0) ? PhysicalIndex(0) : PhysicalIndex(length() - 1) + 1);
412 }

414 // Pre-condition: physical_pos < RunEndsArray(array_span_).length);
415 inline int64_t run_end(int64_t physical_pos) const {
416 assert(physical_pos < RunEndsArray(array_span_).length);
417 // Logical index of the end of the run at physical_pos with offset applied
418 const int64_t logical_run_end =
419 std::max<int64_t>(static_cast<int64_t>(run_ends_[physical_pos]) - offset(), 0);
420 // The current run may go further than the logical length, cap it
421 return std::min(logical_run_end, length());
422 }

425 const ArraySpan& array_span_;

427 const int64_t length_;
428 const int64_t offset_;

436template <typename Left, typename Right>
437class MergedRunsIterator {
438 private:
439 using LeftIterator = typename Left::Iterator;
440 using RightIterator = typename Right::Iterator;
441
442 MergedRunsIterator(LeftIterator left_it, RightIterator right_it,
443 int64_t common_logical_length, int64_t common_logical_pos)
444 : ree_iterators_{std::move(left_it), std::move(right_it)},
445 logical_length_(common_logical_length),
446 logical_pos_(common_logical_pos) {}
447
448 public:
452 MergedRunsIterator(const Left& left, const Right& right)
453 : MergedRunsIterator(left.begin(), right.begin(), left.length(), 0) {
454 assert(left.length() == right.length());
455 }
456
457 static Result<MergedRunsIterator> MakeBegin(const Left& left, const Right& right) {
458 if (left.length() != right.length()) {
459 return Status::Invalid(
460 "MergedRunsIterator expects RunEndEncodedArraySpans of the same length");
461 }
462 return MergedRunsIterator(left, right);
463 }
464
465 static Result<MergedRunsIterator> MakeEnd(const Left& left, const Right& right) {
466 if (left.length() != right.length()) {
467 return Status::Invalid(
468 "MergedRunsIterator expects RunEndEncodedArraySpans of the same length");
469 }
470 return MergedRunsIterator(left.end(), right.end(), left.length(), left.length());
471 }
472
474 const Left& left() const { return std::get<0>(ree_iterators_).span; }
475
477 const Right& right() const { return std::get<1>(ree_iterators_).span; }
478
482 int64_t logical_position() const { return logical_pos_; }
483
485 bool is_begin() const { return logical_pos_ == 0; }
486
488 bool is_end() const { return logical_pos_ == logical_length_; }
489
493 int64_t run_end() const {
494 const auto& left_it = std::get<0>(ree_iterators_);
495 const auto& right_it = std::get<1>(ree_iterators_);
496 return std::min(left_it.run_end(), right_it.run_end());
497 }
498
502 int64_t run_length() const { return run_end() - logical_pos_; }
503
506 template <size_t input_id>
507 int64_t index_into_array() const {
508 return std::get<input_id>(ree_iterators_).index_into_array();
509 }
510
511 int64_t index_into_left_array() const { return index_into_array<0>(); }
512 int64_t index_into_right_array() const { return index_into_array<1>(); }
513
514 MergedRunsIterator& operator++() {
515 auto& left_it = std::get<0>(ree_iterators_);
516 auto& right_it = std::get<1>(ree_iterators_);
517
518 const int64_t left_run_end = left_it.run_end();
519 const int64_t right_run_end = right_it.run_end();
520
521 if (left_run_end < right_run_end) {
522 logical_pos_ = left_run_end;
523 ++left_it;
524 } else if (left_run_end > right_run_end) {
525 logical_pos_ = right_run_end;
526 ++right_it;
527 } else {
528 logical_pos_ = left_run_end;
529 ++left_it;
530 ++right_it;
531 }
532 return *this;
533 }
534
535 MergedRunsIterator operator++(int) {
536 MergedRunsIterator prev = *this;
537 ++(*this);
538 return prev;
539 }
540
541 MergedRunsIterator& operator--() {
542 auto& left_it = std::get<0>(ree_iterators_);
543 auto& right_it = std::get<1>(ree_iterators_);
544
545 // The logical position of each iterator is the run_end() of the previous run.
546 const int64_t left_logical_pos = left_it.logical_position();
547 const int64_t right_logical_pos = right_it.logical_position();
548
549 if (left_logical_pos < right_logical_pos) {
550 --right_it;
551 logical_pos_ = std::max(left_logical_pos, right_it.logical_position());
552 } else if (left_logical_pos > right_logical_pos) {
553 --left_it;
554 logical_pos_ = std::max(left_it.logical_position(), right_logical_pos);
555 } else {
556 --left_it;
557 --right_it;
558 logical_pos_ = std::max(left_it.logical_position(), right_it.logical_position());
559 }
560 return *this;
561 }
562
563 MergedRunsIterator operator--(int) {
564 MergedRunsIterator prev = *this;
565 --(*this);
566 return prev;
567 }
568
569 bool operator==(const MergedRunsIterator& other) const {
570 return logical_pos_ == other.logical_position();
571 }
572
573 bool operator!=(const MergedRunsIterator& other) const { return !(*this == other); }
574
575 private:
576 std::tuple<LeftIterator, RightIterator> ree_iterators_;
577 const int64_t logical_length_;
578 int64_t logical_pos_;
579};
