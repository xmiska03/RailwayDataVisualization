141template <int32_t byte_width>
142static void random_decimals(int64_t n, uint32_t seed, int32_t precision, uint8_t* out) {
143 auto gen = ::arrow::random::RandomArrayGenerator(seed);
144 std::shared_ptr<Array> decimals;
145 if constexpr (byte_width == 4) {
146 decimals = gen.Decimal32(::arrow::decimal32(precision, 0), n);
147 } else if constexpr (byte_width == 8) {
148 decimals = gen.Decimal64(::arrow::decimal64(precision, 0), n);
149 } else if constexpr (byte_width == 16) {
150 decimals = gen.Decimal128(::arrow::decimal128(precision, 0), n);
151 } else {
152 decimals = gen.Decimal256(::arrow::decimal256(precision, 0), n);
153 }
154 std::memcpy(out, decimals->data()->GetValues<uint8_t>(1, 0), byte_width * n);
155}

285template <typename ArrowType>
286::arrow::enable_if_base_binary<ArrowType, Status> NullableArray(
287 size_t size, size_t num_nulls, uint32_t seed, std::shared_ptr<::arrow::Array>* out) {
288 std::vector<uint8_t> valid_bytes(size, 1);
289
290 for (size_t i = 0; i < num_nulls; i++) {
291 valid_bytes[i * 2] = 0;
292 }
293
294 using BuilderType = typename ::arrow::TypeTraits<ArrowType>::BuilderType;
295 BuilderType builder;
296
297 const int kBufferSize = 10;
298 uint8_t buffer[kBufferSize];
299 for (size_t i = 0; i < size; i++) {
300 if (!valid_bytes[i]) {
301 RETURN_NOT_OK(builder.AppendNull());
302 } else {
303 ::arrow::random_bytes(kBufferSize, seed + static_cast<uint32_t>(i), buffer);
304 if (ArrowType::is_utf8) {
305 // Trivially force data to be valid UTF8 by making it all ASCII
306 for (auto& byte : buffer) {
307 byte &= 0x7f;
308 }
309 }
310 RETURN_NOT_OK(builder.Append(buffer, kBufferSize));
311 }
312 }
313 return builder.Finish(out);
314}

423Status MakeListArray(const std::shared_ptr<Array>& values, int64_t size,
424 int64_t null_count, const std::string& item_name,
425 bool nullable_values, std::shared_ptr<::arrow::ListArray>* out) {
426 // We always include an empty list
427 int64_t non_null_entries = size - null_count - 1;
428 int64_t length_per_entry = values->length() / non_null_entries;
429
430 auto offsets = AllocateBuffer();
431 RETURN_NOT_OK(offsets->Resize((size + 1) * sizeof(int32_t)));
432 int32_t* offsets_ptr = reinterpret_cast<int32_t*>(offsets->mutable_data());
433
434 auto null_bitmap = AllocateBuffer();
435 int64_t bitmap_size = ::arrow::bit_util::BytesForBits(size);
436 RETURN_NOT_OK(null_bitmap->Resize(bitmap_size));
437 uint8_t* null_bitmap_ptr = null_bitmap->mutable_data();
438 memset(null_bitmap_ptr, 0, bitmap_size);
439
440 int32_t current_offset = 0;
441 for (int64_t i = 0; i < size; i++) {
442 offsets_ptr[i] = current_offset;
443 if (!(((i % 2) == 0) && ((i / 2) < null_count))) {
444 // Non-null list (list with index 1 is always empty).
445 ::arrow::bit_util::SetBit(null_bitmap_ptr, i);
446 if (i != 1) {
447 current_offset += static_cast<int32_t>(length_per_entry);
448 }
449 }
450 }
451 offsets_ptr[size] = static_cast<int32_t>(values->length());
452
453 auto value_field = ::arrow::field(item_name, values->type(), nullable_values);
454 *out = std::make_shared<::arrow::ListArray>(::arrow::list(value_field), size, offsets,
455 values, null_bitmap, null_count);
456
457 return Status::OK();
458}

460// Make an array containing only empty lists, with a null values array
461Status MakeEmptyListsArray(int64_t size, std::shared_ptr<Array>* out_array) {
462 // Allocate an offsets buffer containing only zeroes
463 const int64_t offsets_nbytes = (size + 1) * sizeof(int32_t);
464 ARROW_ASSIGN_OR_RAISE(auto offsets_buffer, ::arrow::AllocateBuffer(offsets_nbytes));
465 memset(offsets_buffer->mutable_data(), 0, offsets_nbytes);
466
467 auto value_field =
468 ::arrow::field("item", ::arrow::float64(), false /* nullable_values */);
469 auto list_type = ::arrow::list(value_field);
470
471 std::vector<std::shared_ptr<Buffer>> child_buffers = {nullptr /* null bitmap */,
472 nullptr /* values */};
473 auto child_data =
474 ::arrow::ArrayData::Make(value_field->type(), 0, std::move(child_buffers));
475
476 std::vector<std::shared_ptr<Buffer>> buffers = {nullptr /* bitmap */,
477 std::move(offsets_buffer)};
478 auto array_data = ::arrow::ArrayData::Make(list_type, size, std::move(buffers));
479 array_data->child_data.push_back(child_data);
480
481 *out_array = ::arrow::MakeArray(array_data);
482 return Status::OK();
483}
