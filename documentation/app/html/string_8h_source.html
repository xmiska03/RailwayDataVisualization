126template <typename T>
127inline constexpr bool have_to_chars = detail::can_to_chars<T>::value;
128
136template <typename T, typename... Args>
137std::string ToChars(T value, Args&&... args) {
138 if constexpr (!have_to_chars<T>) {
139 // Some C++ standard libraries do not yet implement std::to_chars for all types,
140 // in which case we have to fallback to std::string.
141 return std::to_string(value);
142 } else {
143 // According to various sources, the GNU libstdc++ and Microsoft's C++ STL
144 // allow up to 15 bytes of small string optimization, while clang's libc++
145 // goes up to 22 bytes. Choose the pessimistic value.
146 std::string out(15, 0);
147 auto res = std::to_chars(&out.front(), &out.back(), value, args...);
148 while (res.ec != std::errc{}) {
149 assert(res.ec == std::errc::value_too_large);
150 out.resize(out.capacity() * 2);
151 res = std::to_chars(&out.front(), &out.back(), value, args...);
152 }
153 const auto length = res.ptr - out.data();
154 assert(length <= static_cast<int64_t>(out.length()));
155 out.resize(length);
156 return out;
157 }
158}
