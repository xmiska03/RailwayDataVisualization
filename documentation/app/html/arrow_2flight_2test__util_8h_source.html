46inline void AssertEqual(const FlightInfo& expected, const FlightInfo& actual) {
47 ipc::DictionaryMemo expected_memo;
48 ipc::DictionaryMemo actual_memo;
49 ASSERT_OK_AND_ASSIGN(auto ex_schema, expected.GetSchema(&expected_memo));
50 ASSERT_OK_AND_ASSIGN(auto actual_schema, actual.GetSchema(&actual_memo));
51
52 AssertSchemaEqual(*ex_schema, *actual_schema);
53 ASSERT_EQ(expected.total_records(), actual.total_records());
54 ASSERT_EQ(expected.total_bytes(), actual.total_bytes());
55
56 ASSERT_EQ(expected.descriptor(), actual.descriptor());
57 ASSERT_THAT(actual.endpoints(), ::testing::ContainerEq(expected.endpoints()));
58}

89// Helper to initialize a server and matching client with callbacks to
90// populate options.
91template <typename T, typename... Args>
92Status MakeServer(const Location& location, std::unique_ptr<FlightServerBase>* server,
93 std::unique_ptr<FlightClient>* client,
94 std::function<Status(FlightServerOptions*)> make_server_options,
95 std::function<Status(FlightClientOptions*)> make_client_options,
96 Args&&... server_args) {
97 *server = std::make_unique<T>(std::forward<Args>(server_args)...);
98 FlightServerOptions server_options(location);
99 RETURN_NOT_OK(make_server_options(&server_options));
100 RETURN_NOT_OK((*server)->Init(server_options));
101 std::string uri =
102 location.scheme() + "://127.0.0.1:" + std::to_string((*server)->port());
103 ARROW_ASSIGN_OR_RAISE(auto real_location, Location::Parse(uri));
104 FlightClientOptions client_options = FlightClientOptions::Defaults();
105 RETURN_NOT_OK(make_client_options(&client_options));
106 return FlightClient::Connect(real_location, client_options).Value(client);
107}

109// Helper to initialize a server and matching client with callbacks to
110// populate options.
111template <typename T, typename... Args>
112Status MakeServer(std::unique_ptr<FlightServerBase>* server,
113 std::unique_ptr<FlightClient>* client,
114 std::function<Status(FlightServerOptions*)> make_server_options,
115 std::function<Status(FlightClientOptions*)> make_client_options,
116 Args&&... server_args) {
117 ARROW_ASSIGN_OR_RAISE(auto location, Location::ForGrpcTcp("localhost", 0));
118 return MakeServer<T>(location, server, client, std::move(make_server_options),
119 std::move(make_client_options),
120 std::forward<Args>(server_args)...);
121}
