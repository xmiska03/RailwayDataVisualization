31// A parallelizer that takes a `Status(int)` function and calls it with
32// arguments between 0 and `num_tasks - 1`, on an arbitrary number of threads.
33
34template <class FUNCTION>
35Status ParallelFor(int num_tasks, FUNCTION&& func,
36 Executor* executor = internal::GetCpuThreadPool()) {
37 std::vector<Future<>> futures(num_tasks);
38
39 for (int i = 0; i < num_tasks; ++i) {
40 ARROW_ASSIGN_OR_RAISE(futures[i], executor->Submit(func, i));
41 }
42 auto st = Status::OK();
43 for (auto& fut : futures) {
44 st &= fut.status();
45 }
46 return st;
47}

52 std::vector<T> inputs, FUNCTION&& func,
53 Executor* executor = internal::GetCpuThreadPool()) {

55 for (size_t i = 0; i < inputs.size(); ++i) {
56 ARROW_ASSIGN_OR_RAISE(futures[i], executor->Submit(func, i, std::move(inputs[i])));
57 }
58 return All(std::move(futures))
59 .Then([](const std::vector<Result<R>>& results) -> Result<std::vector<R>> {
60 return UnwrapOrRaise(results);
61 });

64// A parallelizer that takes a `Status(int)` function and calls it with
65// arguments between 0 and `num_tasks - 1`, in sequence or in parallel,
66// depending on the input boolean.
67
68template <class FUNCTION>
69Status OptionalParallelFor(bool use_threads, int num_tasks, FUNCTION&& func,
70 Executor* executor = internal::GetCpuThreadPool()) {
71 if (use_threads) {
72 return ParallelFor(num_tasks, std::forward<FUNCTION>(func), executor);
73 } else {
74 for (int i = 0; i < num_tasks; ++i) {
75 RETURN_NOT_OK(func(i));
76 }
77 return Status::OK();
78 }
79}

81// A parallelizer that takes a `Result<R>(int index, T item)` function and
82// calls it with each item from the input array, in sequence or in parallel,
83// depending on the input boolean.
84
85template <class FUNCTION, typename T,
86 typename R = typename internal::call_traits::return_type<FUNCTION>::ValueType>
87Future<std::vector<R>> OptionalParallelForAsync(
88 bool use_threads, std::vector<T> inputs, FUNCTION&& func,
89 Executor* executor = internal::GetCpuThreadPool()) {
90 if (use_threads) {
91 return ParallelForAsync(std::move(inputs), std::forward<FUNCTION>(func), executor);
92 } else {
93 std::vector<R> result(inputs.size());
94 for (size_t i = 0; i < inputs.size(); ++i) {
95 ARROW_ASSIGN_OR_RAISE(result[i], func(i, inputs[i]));
96 }
97 return result;
98 }
