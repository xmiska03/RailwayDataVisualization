194// API to write values to a single column. This is the main client facing API.
195template <typename DType>
196class TypedColumnWriter : public ColumnWriter {
197 public:
198 using T = typename DType::c_type;
199
200 // Write a batch of repetition levels, definition levels, and values to the
201 // column.
202 // `num_values` is the number of logical leaf values.
203 // `def_levels` (resp. `rep_levels`) can be null if the column's max definition level
204 // (resp. max repetition level) is 0.
205 // If not null, each of `def_levels` and `rep_levels` must have at least

258constexpr int64_t kJulianEpochOffsetDays = INT64_C(2440588);
259
260template <int64_t UnitPerDay, int64_t NanosecondsPerUnit>
261inline void ArrowTimestampToImpalaTimestamp(const int64_t time, Int96* impala_timestamp) {
262 int64_t julian_days = (time / UnitPerDay) + kJulianEpochOffsetDays;
263 (*impala_timestamp).value[2] = (uint32_t)julian_days;
264
265 int64_t last_day_units = time % UnitPerDay;
266 auto last_day_nanos = last_day_units * NanosecondsPerUnit;
267 // impala_timestamp will be unaligned every other entry so do memcpy instead
268 // of assign and reinterpret cast to avoid undefined behavior.
269 std::memcpy(impala_timestamp, &last_day_nanos, sizeof(int64_t));
270}
