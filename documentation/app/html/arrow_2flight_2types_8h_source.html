181// Base CRTP type
182template <class T>
183struct BaseType {
184 protected:
185 using SuperT = BaseType<T>;
186 using SelfT = typename remove_unique_ptr<T>::type;
187
188 const SelfT& self() const { return static_cast<const SelfT&>(*this); }
189 SelfT& self() { return static_cast<SelfT&>(*this); }
190
191 public:
192 BaseType() = default;
193
194 friend bool operator==(const SelfT& left, const SelfT& right) {
195 return left.Equals(right);
196 }
197 friend bool operator!=(const SelfT& left, const SelfT& right) {
198 return !left.Equals(right);
199 }
200
202 inline arrow::Result<std::string> SerializeToString() const {
203 std::string out;
204 ARROW_RETURN_NOT_OK(self().SelfT::SerializeToString(&out));
205 return out;
206 }
207
208 inline static arrow::Result<T> Deserialize(std::string_view serialized) {
209 T out;
210 ARROW_RETURN_NOT_OK(SelfT::Deserialize(serialized, &out));
211 return out;
212 }
213
214 inline arrow::Result<std::shared_ptr<Buffer>> SerializeToBuffer() const {
215 std::string out;
216 ARROW_RETURN_NOT_OK(self().SelfT::SerializeToString(&out));
217 return Buffer::FromString(std::move(out));
218 }
219};
