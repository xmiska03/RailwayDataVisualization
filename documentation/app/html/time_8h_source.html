52template <typename Visitor, typename... Args>
53decltype(std::declval<Visitor>()(std::chrono::seconds{}, std::declval<Args&&>()...))
54VisitDuration(TimeUnit::type unit, Visitor&& visitor, Args&&... args) {
55 switch (unit) {
56 default:
57 case TimeUnit::SECOND:
58 break;
59 case TimeUnit::MILLI:
60 return visitor(std::chrono::milliseconds{}, std::forward<Args>(args)...);
61 case TimeUnit::MICRO:
62 return visitor(std::chrono::microseconds{}, std::forward<Args>(args)...);
63 case TimeUnit::NANO:
64 return visitor(std::chrono::nanoseconds{}, std::forward<Args>(args)...);
65 }
66 return visitor(std::chrono::seconds{}, std::forward<Args>(args)...);
67}

70struct CastSecondsToUnitImpl {
71 template <typename Duration>
72 int64_t operator()(Duration, int64_t seconds) {
73 auto duration = std::chrono::duration_cast<Duration>(std::chrono::seconds{seconds});
74 return static_cast<int64_t>(duration.count());
75 }
76};
77
78inline int64_t CastSecondsToUnit(TimeUnit::type unit, int64_t seconds) {
79 return VisitDuration(unit, CastSecondsToUnitImpl{}, seconds);
80}
