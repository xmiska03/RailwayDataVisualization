619// Julian day at unix epoch.
620//
621// The Julian Day Number (JDN) is the integer assigned to a whole solar day in
622// the Julian day count starting from noon Universal time, with Julian day
623// number 0 assigned to the day starting at noon on Monday, January 1, 4713 BC,
624// proleptic Julian calendar (November 24, 4714 BC, in the proleptic Gregorian
625// calendar),
626constexpr int64_t kJulianToUnixEpochDays = INT64_C(2440588);
627constexpr int64_t kSecondsPerDay = INT64_C(60 * 60 * 24);
628constexpr int64_t kMillisecondsPerDay = kSecondsPerDay * INT64_C(1000);
629constexpr int64_t kMicrosecondsPerDay = kMillisecondsPerDay * INT64_C(1000);
630constexpr int64_t kNanosecondsPerDay = kMicrosecondsPerDay * INT64_C(1000);
631
632MANUALLY_ALIGNED_STRUCT(1) Int96 { uint32_t value[3]; };
633STRUCT_END(Int96, 12);

636 return std::equal(left.value, left.value + 3, right.value);
637}

639inline bool operator!=(const Int96& left, const Int96& right) { return !(left == right); }

649struct DecodedInt96 {
650 uint64_t days_since_epoch;
651 uint64_t nanoseconds;
652};
653
654static inline DecodedInt96 DecodeInt96Timestamp(const parquet::Int96& i96) {
655 // We do the computations in the unsigned domain to avoid unsigned behaviour
656 // on overflow.
657 DecodedInt96 result;
658 result.days_since_epoch = i96.value[2] - static_cast<uint64_t>(kJulianToUnixEpochDays);
659 result.nanoseconds = 0;
660
661 memcpy(&result.nanoseconds, &i96.value, sizeof(uint64_t));
662 return result;
663}
664
665static inline int64_t Int96GetNanoSeconds(const parquet::Int96& i96) {
666 const auto decoded = DecodeInt96Timestamp(i96);
667 return static_cast<int64_t>(decoded.days_since_epoch * kNanosecondsPerDay +
668 decoded.nanoseconds);
669}
670
671static inline int64_t Int96GetMicroSeconds(const parquet::Int96& i96) {
672 const auto decoded = DecodeInt96Timestamp(i96);
673 uint64_t microseconds = decoded.nanoseconds / static_cast<uint64_t>(1000);
674 return static_cast<int64_t>(decoded.days_since_epoch * kMicrosecondsPerDay +
675 microseconds);
676}
677
678static inline int64_t Int96GetMilliSeconds(const parquet::Int96& i96) {
679 const auto decoded = DecodeInt96Timestamp(i96);
680 uint64_t milliseconds = decoded.nanoseconds / static_cast<uint64_t>(1000000);
681 return static_cast<int64_t>(decoded.days_since_epoch * kMillisecondsPerDay +
682 milliseconds);
683}
684
685static inline int64_t Int96GetSeconds(const parquet::Int96& i96) {
686 const auto decoded = DecodeInt96Timestamp(i96);
687 uint64_t seconds = decoded.nanoseconds / static_cast<uint64_t>(1000000000);
688 return static_cast<int64_t>(decoded.days_since_epoch * kSecondsPerDay + seconds);
689}
