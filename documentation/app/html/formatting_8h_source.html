137// A 2x100 direct table mapping integers in [0..99] to their decimal representations.
138ARROW_EXPORT extern const char digit_pairs[];
139
140// Based on fmtlib's format_int class:
141// Write digits from right to left into a stack allocated buffer.
142// \pre *cursor points to the byte after the one that will be written.
143// \post *cursor points to the byte that was written.
144inline void FormatOneChar(char c, char** cursor) { *(--(*cursor)) = c; }
145
146template <typename Int>
147void FormatOneDigit(Int value, char** cursor) {
148 assert(value >= 0 && value <= 9);
149 FormatOneChar(static_cast<char>('0' + value), cursor);
150}
151
152// GH-35662: I don't know why but the following combination causes SEGV:
153// * template implementation without inline
154// * MinGW
155// * Release build
156template <typename Int>
157inline void FormatTwoDigits(Int value, char** cursor) {
158 assert(value >= 0 && value <= 99);
159 auto digit_pair = &digit_pairs[value * 2];
160 FormatOneChar(digit_pair[1], cursor);
161 FormatOneChar(digit_pair[0], cursor);
162}
163
164template <typename Int>
165void FormatAllDigits(Int value, char** cursor) {
166 assert(value >= 0);
167 while (value >= 100) {
168 FormatTwoDigits(value % 100, cursor);
169 value /= 100;
170 }
171
172 if (value >= 10) {
173 FormatTwoDigits(value, cursor);
174 } else {
175 FormatOneDigit(value, cursor);
176 }
177}
178
179template <typename Int>
180void FormatAllDigitsLeftPadded(Int value, size_t pad, char pad_char, char** cursor) {
181 auto end = *cursor - pad;
182 FormatAllDigits(value, cursor);
183 while (*cursor > end) {
184 FormatOneChar(pad_char, cursor);
185 }
186}
187
188template <size_t BUFFER_SIZE>
189std::string_view ViewDigitBuffer(const std::array<char, BUFFER_SIZE>& buffer,
190 char* cursor) {
191 auto buffer_end = buffer.data() + BUFFER_SIZE;
192 return {cursor, static_cast<size_t>(buffer_end - cursor)};
193}
194
195template <typename Int, typename UInt = typename std::make_unsigned<Int>::type>
196constexpr UInt Abs(Int value) {
197 return value < 0 ? ~static_cast<UInt>(value) + 1 : static_cast<UInt>(value);
198}
199
200template <typename Int>
201constexpr size_t Digits10(Int value) {
202 return value <= 9 ? 1 : Digits10(value / 10) + 1;
203}

342constexpr size_t BufferSizeYYYY_MM_DD() {
343 // "-"? "99999-12-31"
344 return 1 + detail::Digits10(99999) + 1 + detail::Digits10(12) + 1 +
345 detail::Digits10(31);
346}
347
348inline void FormatYYYY_MM_DD(arrow_vendored::date::year_month_day ymd, char** cursor) {
349 FormatTwoDigits(static_cast<unsigned>(ymd.day()), cursor);
350 FormatOneChar('-', cursor);
351 FormatTwoDigits(static_cast<unsigned>(ymd.month()), cursor);
352 FormatOneChar('-', cursor);
353 auto year = static_cast<int>(ymd.year());
354 const auto is_neg_year = year < 0;
355 year = std::abs(year);
356 assert(year <= 99999);
357 FormatTwoDigits(year % 100, cursor);
358 year /= 100;
359 FormatTwoDigits(year % 100, cursor);
360 if (year >= 100) {
361 FormatOneDigit(year / 100, cursor);
362 }
363 if (is_neg_year) {
364 FormatOneChar('-', cursor);
365 }
366}

368template <typename Duration>
369constexpr size_t BufferSizeHH_MM_SS() {
370 // "23:59:59" ("." "9"+)?
371 return detail::Digits10(23) + 1 + detail::Digits10(59) + 1 + detail::Digits10(59) + 1 +
372 detail::Digits10(Duration::period::den) - 1;
373}
374
375template <typename Duration>
376void FormatHH_MM_SS(arrow_vendored::date::hh_mm_ss<Duration> hms, char** cursor) {
377 constexpr size_t subsecond_digits = Digits10(Duration::period::den) - 1;
378 if (subsecond_digits != 0) {
379 FormatAllDigitsLeftPadded(hms.subseconds().count(), subsecond_digits, '0', cursor);
380 FormatOneChar('.', cursor);
381 }
382 FormatTwoDigits(hms.seconds().count(), cursor);
383 FormatOneChar(':', cursor);
384 FormatTwoDigits(hms.minutes().count(), cursor);
385 FormatOneChar(':', cursor);
386 FormatTwoDigits(hms.hours().count(), cursor);
387}

389// Some out-of-bound datetime values would result in erroneous printing
390// because of silent integer wraparound in the `arrow_vendored::date` library.
391//
392// To avoid such misprinting, we must therefore check the bounds explicitly.
393// The bounds correspond to start of year -32767 and end of year 32767,
394// respectively (-32768 is an invalid year value in `arrow_vendored::date`).
395//
396// Note these values are the same as documented for C++20:
397// https://en.cppreference.com/w/cpp/chrono/year_month_day/operator_days
398template <typename Unit>
399bool IsDateTimeInRange(Unit duration) {
400 constexpr Unit kMinIncl =
401 std::chrono::duration_cast<Unit>(arrow_vendored::date::days{-12687428});
402 constexpr Unit kMaxExcl =
403 std::chrono::duration_cast<Unit>(arrow_vendored::date::days{11248738});
404 return duration >= kMinIncl && duration < kMaxExcl;
405}
406
407// IsDateTimeInRange() specialization for nanoseconds: a 64-bit number of
408// nanoseconds cannot represent years outside of the [-32767, 32767]
409// range, and the {kMinIncl, kMaxExcl} constants above would overflow.
410constexpr bool IsDateTimeInRange(std::chrono::nanoseconds duration) { return true; }
411
412template <typename Unit>
413bool IsTimeInRange(Unit duration) {
414 constexpr Unit kMinIncl = std::chrono::duration_cast<Unit>(std::chrono::seconds{0});
415 constexpr Unit kMaxExcl = std::chrono::duration_cast<Unit>(std::chrono::seconds{86400});
416 return duration >= kMinIncl && duration < kMaxExcl;
417}

419template <typename RawValue, typename Appender>
420Return<Appender> FormatOutOfRange(RawValue&& raw_value, Appender&& append) {
421 // XXX locale-sensitive but good enough for now
422 std::string formatted = "<value out of range: " + ToChars(raw_value) + ">";
423 return append(std::move(formatted));
424}

490template <>
491class StringFormatter<TimestampType> {
492 public:
493 using value_type = int64_t;
494
495 explicit StringFormatter(const DataType* type)
496 : unit_(checked_cast<const TimestampType&>(*type).unit()),
497 timezone_(checked_cast<const TimestampType&>(*type).timezone()) {}
498
499 template <typename Duration, typename Appender>
500 Return<Appender> operator()(Duration, value_type value, Appender&& append) {
501 using arrow_vendored::date::days;
502
503 const Duration since_epoch{value};
504 if (!ARROW_PREDICT_TRUE(detail::IsDateTimeInRange(since_epoch))) {
505 return detail::FormatOutOfRange(value, append);
506 }
507
508 const auto timepoint = detail::kEpoch + since_epoch;
509 // Round days towards zero
510 // (the naive approach of using arrow_vendored::date::floor() would
511 // result in UB for very large negative timestamps, similarly as
512 // https://github.com/HowardHinnant/date/issues/696)
513 auto timepoint_days = std::chrono::time_point_cast<days>(timepoint);
514 Duration since_midnight;
515 if (timepoint_days <= timepoint) {
516 // Year >= 1970
517 since_midnight = timepoint - timepoint_days;
518 } else {
519 // Year < 1970
520 since_midnight = days(1) - (timepoint_days - timepoint);
521 timepoint_days -= days(1);
522 }
523
524 // YYYY_MM_DD " " HH_MM_SS "Z"?
525 constexpr size_t buffer_size =
526 detail::BufferSizeYYYY_MM_DD() + 1 + detail::BufferSizeHH_MM_SS<Duration>() + 1;
527
528 std::array<char, buffer_size> buffer;
529 char* cursor = buffer.data() + buffer_size;
530
531 if (timezone_.size() > 0) {
532 detail::FormatOneChar('Z', &cursor);
533 }
534 detail::FormatHH_MM_SS(arrow_vendored::date::make_time(since_midnight), &cursor);
535 detail::FormatOneChar(' ', &cursor);
536 detail::FormatYYYY_MM_DD(timepoint_days, &cursor);
537 return append(detail::ViewDigitBuffer(buffer, cursor));
538 }

540 template <typename Appender>
541 Return<Appender> operator()(value_type value, Appender&& append) {
542 return util::VisitDuration(unit_, *this, value, std::forward<Appender>(append));
543 }

545 private:
546 TimeUnit::type unit_;
547 std::string timezone_;
548};
