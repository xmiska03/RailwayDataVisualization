42
44 Status AppendNulls(int64_t length) final {

102 Status Append(const value_type val) {
103 ARROW_RETURN_NOT_OK(ArrayBuilder::Reserve(1));

105 return Status::OK();

107
111 Status AppendNulls(int64_t length) final {
112 ARROW_RETURN_NOT_OK(Reserve(length));
113 data_builder_.UnsafeAppend(length, value_type{}); // zero
114 UnsafeSetNull(length);
115 return Status::OK();
116 }
117
119 Status AppendNull() final {
120 ARROW_RETURN_NOT_OK(Reserve(1));
121 data_builder_.UnsafeAppend(value_type{}); // zero
122 UnsafeAppendToBitmap(false);
123 return Status::OK();
124 }
125
127 Status AppendEmptyValue() final {
128 ARROW_RETURN_NOT_OK(Reserve(1));
129 data_builder_.UnsafeAppend(value_type{}); // zero
130 UnsafeAppendToBitmap(true);
131 return Status::OK();
132 }
133
135 Status AppendEmptyValues(int64_t length) final {
136 ARROW_RETURN_NOT_OK(Reserve(length));
137 data_builder_.UnsafeAppend(length, value_type{}); // zero
138 UnsafeSetNotNull(length);
139 return Status::OK();
140 }
141
142 value_type GetValue(int64_t index) const { return data_builder_.data()[index]; }
143
144 void Reset() override {
145 data_builder_.Reset();
146 ArrayBuilder::Reset();
147 }
148
149 Status Resize(int64_t capacity) override {
150 ARROW_RETURN_NOT_OK(CheckCapacity(capacity));
151 capacity = std::max(capacity, kMinBuilderCapacity);
152 ARROW_RETURN_NOT_OK(data_builder_.Resize(capacity));
153 return ArrayBuilder::Resize(capacity);
154 }
155
156 value_type operator[](int64_t index) const { return GetValue(index); }
157
158 value_type& operator[](int64_t index) {
159 return reinterpret_cast<value_type*>(data_builder_.mutable_data())[index];
160 }
161
168 Status AppendValues(const value_type* values, int64_t length,
169 const uint8_t* valid_bytes = NULLPTR) {
170 ARROW_RETURN_NOT_OK(Reserve(length));
171 data_builder_.UnsafeAppend(values, length);
172 // length_ is update by these
173 ArrayBuilder::UnsafeAppendToBitmap(valid_bytes, length);
174 return Status::OK();
175 }

190 }

198 Status AppendValues(const value_type* values, int64_t length,

224 Status FinishInternal(std::shared_ptr<ArrayData>* out) override {
225 ARROW_ASSIGN_OR_RAISE(auto null_bitmap,
226 null_bitmap_builder_.FinishWithLength(length_));
227 ARROW_ASSIGN_OR_RAISE(auto data, data_builder_.FinishWithLength(length_));
228 *out = ArrayData::Make(type(), length_, {null_bitmap, data}, null_count_);
229 capacity_ = length_ = null_count_ = 0;
230 return Status::OK();
231 }

245 int64_t length = static_cast<int64_t>(std::distance(values_begin, values_end));
246 ARROW_RETURN_NOT_OK(Reserve(length));

259 template <typename ValuesIter, typename ValidIter>
260 enable_if_t<!std::is_pointer<ValidIter>::value, Status> AppendValues(
261 ValuesIter values_begin, ValuesIter values_end, ValidIter valid_begin) {
262 static_assert(!internal::is_null_pointer<ValidIter>::value,
263 "Don't pass a NULLPTR directly as valid_begin, use the 2-argument "
264 "version instead");
265 int64_t length = static_cast<int64_t>(std::distance(values_begin, values_end));
266 ARROW_RETURN_NOT_OK(Reserve(length));
267 data_builder_.UnsafeAppend(values_begin, values_end);
268 null_bitmap_builder_.UnsafeAppend<true>(
269 length, [&valid_begin]() -> bool { return *valid_begin++; });
270 length_ = null_bitmap_builder_.length();
271 null_count_ = null_bitmap_builder_.false_count();
272 return Status::OK();
273 }
