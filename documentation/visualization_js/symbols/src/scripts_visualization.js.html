 77 /**
 78  * Maps point intensity to a color on the blue-green-red color scale.
 79  * @param {number[]} d - The point data in format [x, y, z, intensity].
 80  */
 81 function getColorBGR(d) {
 82   if (d[3] < window.scale_boundaries[0]) {
 83     return [0, 0, 255]   // under the scale - blue
 84   } else if (d[3] > window.scale_boundaries[2]) {
 85     return [255, 0, 0]   // over the scale - red
 86   } else if (d[3] < window.scale_boundaries[1]) {   // on the scale
 87     return [
 88       0, 
 89       Math.floor(255 * (d[3] - window.scale_boundaries[0]) 
 90                   / (window.scale_boundaries[1] - window.scale_boundaries[0])),
 91       Math.floor(255 - 255 * (d[3] - window.scale_boundaries[0]) 
 92                         / (window.scale_boundaries[1] - window.scale_boundaries[0]))
 93     ]
 94   } else {
 95     return [
 96       Math.floor(255 * (d[3] - window.scale_boundaries[1]) 
 97                   / (window.scale_boundaries[2] - window.scale_boundaries[1])),
 98       Math.floor(255 - 255 * (d[3] - window.scale_boundaries[1])
 99                         / (window.scale_boundaries[2] - window.scale_boundaries[1])),
100       0
101     ]
102   }
103 }

106 /**
107  * Maps point intensity to a color on the yellow-purple color scale.
108  * @param {number[]} d - The point data in format [x, y, z, intensity].
109  */
110 function getColorYP(d) {
111   if (d[3] < window.scale_boundaries[0]) {
112     return [255, 255, 0]   // under the scale - yellow
113   } else if (d[3] > window.scale_boundaries[2]) {
114     return [255, 0, 255]   // over the scale - purple
115   } else {     // on the scale
116     return [
117       255,
118       Math.floor(255 - 255 * (d[3] - window.scale_boundaries[0]) 
119                         / (window.scale_boundaries[2] - window.scale_boundaries[0])),
120       Math.floor(255 * (d[3] - window.scale_boundaries[0]) 
121                   / (window.scale_boundaries[2] - window.scale_boundaries[0]))
122     ]
123   }
124 }

127 /**
128  * Converts train profile distance in meters to an index in the paths data and transformations data.
129  * @param {string} profile_dist - Either '25', '50', '75' or '100'.
130  */
131 function profile_distance_to_index(profile_dist) {
132   let i = 0;  // default is '25'
133   switch (window.profile_distance) {
134     case '50':
135       i = 1;
136       break;
137     case '75':
138       i = 2;
139       break;
140     case '100':
141       i = 3;
142       break;
143   }
144   return i;
145 }

148 /**
149  * Applies transformations to the train profile.
150  * A data accessor for the train profile layer.
151  * @param {Array<[number, number, number]>} d - The shape of the train profile (an array of points).
152  */
153 function profileGetPath(d) {
154   const n = profile_distance_to_index(window.profile_distance);
155   
156   let pos = window.position;
157   if (pos > window.profile_transf[n].length - 1) pos = window.profile_transf[n].length - 1;
158   
159   const points = new Array(d.length);
160   for (let i = 0; i < d.length; i++) {
161     const x = d[i][0] * window.profile_transf[n][pos][0][0] + d[i][1] * window.profile_transf[n][pos][0][1] + d[i][2] * window.profile_transf[n][pos][0][2] + window.profile_transf[n][pos][0][3];
162     const y = d[i][0] * window.profile_transf[n][pos][1][0] + d[i][1] * window.profile_transf[n][pos][1][1] + d[i][2] * window.profile_transf[n][pos][1][2] + window.profile_transf[n][pos][1][3];
163     const z = d[i][0] * window.profile_transf[n][pos][2][0] + d[i][1] * window.profile_transf[n][pos][2][1] + d[i][2] * window.profile_transf[n][pos][2][2] + window.profile_transf[n][pos][2][3];
164 
165     points[i] = [x, y, z];
166   }
167   return points;
168 }

171 /**
172  * Creates a point cloud layer for the deck.gl visualization.
173  * @param {number} n - The index of the layer.
174  */
175 function createPointCloudLayer(n) {
176   return new PointCloudLayer({
177     id: `point-cloud-layer${n}`,
178     data: window.display_united
179       ? window.united_pc_data
180       : window.data_dict.layers[0].data[pcl_layers_positions[n]],
181     getColor: window.data_dict.layers[0].pointColor === 'bgr'
182       ? getColorBGR
183       : getColorYP,
184     getPosition: (d) => d,
185     opacity: window.data_dict.layers[0].opacity,
186     pointSize: window.data_dict.layers[0].pointSize,
187     visible: window.data_dict.layers[0].visible,
188     updateTriggers: {      // needed when changing getPosition or getColor (data accessors)
189       getColor: [window.scale_boundaries[0], window.scale_boundaries[2], 
190                  window.data_dict.layers[0].pointColor]
191     }
192   });
193 }

254 /**
255  * Creates all the layers for the deck.gl visualization.
256  */
257 function createLayers() {
258   window.layers = new Array(window.curr_pcl_layers_cnt + 2);
259 
260   for (let i = 0; i < window.curr_pcl_layers_cnt; i++) {
261     window.layers[i] = createPointCloudLayer(i);
262   }
263   window.layers[window.curr_pcl_layers_cnt] = createProfileLineLayer();
264   window.layers[window.curr_pcl_layers_cnt + 1] = createProfileLayer();
265   window.layers[window.curr_pcl_layers_cnt + 2] = createVectorLayer();
266 }

279 /**
280  * Initializes the deck.gl visualization.
281  */
282 function initializeDeck() {
283 
284   if (window.profile_transf == null) {
285     // transformation data was not yet defined by the callback, wait until it is
286     setTimeout(initializeDeck, 40);  // try again in 40 ms
287     return;
288   }
289   
290   const VIEW = new FirstPersonView({
291     projectionMatrix: window.data_dict.views[0].projectionMatrix,
292     controller: window.data_dict.views[0].controller
293   });
294 
295   // The context is created manually to specify "preserveDrawingBuffer: true".
296   // That is needed to enable reading the pixels of the visualisation for applying distortion.
297   const canvas = document.getElementById("visualization-canvas");
298   const context = canvas.getContext("webgl2", { preserveDrawingBuffer: true, premultipliedAlpha: false });
299 
300   window.deck = new Deck({
301     views: [VIEW],
302     layers: [],
303     canvas: 'visualization-canvas',
304     context: context,
305     onDeviceInitialized: () => {
306       // check whether WegGL is HW accelerated
307       let renderer = window.deck.device.info.renderer;
308       // according to this page, the software-based renderers are llvmpipe and SwiftShader
309       // https://deviceandbrowserinfo.com/learning_zone/articles/webgl_renderer_values
310       if (renderer.includes('llvmpipe') || renderer.includes('SwiftShader')) {
311         window.alert(
312           'V tomto prohlížeči není přístupná hardwarová akcelerace grafiky. '
313           + 'Vykreslování většího množství dat bude probíhat pomalu.'
314         )
315       }
316     }
317   });
318 
319   updateDeck();
320   window.deck_initialized = true;
321 }

323 /**
324  * Calculates which point cloud chunks should be displayed at the current position.
325  */
326 function changeLayersData() {
327   if (!window.display_united) {
328     // find current position in point cloud timestamps
329     window.pcl_position = 0;
330     const video = document.getElementById('background-video'); 
331     while(window.pcl_timestamps[window.pcl_position] < video.currentTime) {
332       window.pcl_position++;
333     }
334 
335     // display point cloud data corresponding to current position and 9 positions backwards (if they exist)
336     // for example if window.pcl_position == 7, pcl_layers_position will be [0, 0, 0, 1, 2, 3, 4, 5, 6, 7]
337     let pos = window.pcl_position;
338     for (let i = window.curr_pcl_layers_cnt - 1; i >= 0; i--) {
339       pcl_layers_positions[i] = pos;
340       if (pos > 0) {
341         pos--;
342       }
343     }
344     // the first set of data in the buffer will now be the oldest to rewrite in case of animation start
345     pcl_layers_index = 0;
346   }
347 }

349 /**
350  * Changes camera position.
351  * The final position of the virtual camera is a combination of:
352    1. the current position of the train (determined by the window.position variable, data is in arrays 
353       window.translations, window.rotations_inv, window.rotations_euler),
354    2. custom camera offset set by the user (window.camera_offset).
355  */
356 function updateDeck() {
357   const pos = window.position;
358 
359   const offset_x = window.camera_offset[0];
360   const offset_y = window.camera_offset[1];
361   const offset_z = window.camera_offset[2];
362   const sum_x = offset_x + window.rotations[pos][0][0] * window.translations[pos][0] + window.rotations[pos][0][1] * window.translations[pos][1] + window.rotations[pos][0][2] * window.translations[pos][2];
363   const sum_y = offset_y + window.rotations[pos][1][0] * window.translations[pos][0] + window.rotations[pos][1][1] * window.translations[pos][1] + window.rotations[pos][1][2] * window.translations[pos][2];
364   const sum_z = offset_z + window.rotations[pos][2][0] * window.translations[pos][0] + window.rotations[pos][2][1] * window.translations[pos][1] + window.rotations[pos][2][2] * window.translations[pos][2];
365 
366   // multiply sum_* by inverse rotation matrix
367   const final_x = window.rotations_inv[pos][0][0] * sum_x + window.rotations_inv[pos][0][1] * sum_y + window.rotations_inv[pos][0][2] * sum_z;
368   const final_y = window.rotations_inv[pos][1][0] * sum_x + window.rotations_inv[pos][1][1] * sum_y + window.rotations_inv[pos][1][2] * sum_z;
369   const final_z = window.rotations_inv[pos][2][0] * sum_x + window.rotations_inv[pos][2][1] * sum_y + window.rotations_inv[pos][2][2] * sum_z;
370 
371   // make a new viewstate from the new position
372   const INITIAL_VIEW_STATE = {
373     bearing: 90 + window.rotations_euler[pos][0] + window.camera_offset[4],
374     pitch: window.rotations_euler[pos][1] + window.camera_offset[5],
375     position: [final_x, final_y, final_z]
376   };
377   window.deck.setProps({initialViewState: INITIAL_VIEW_STATE});
378 
379   // add roll angle by rotating the canvas
380   const canvas = document.getElementById('visualization-canvas');
381   const dist_canvas = document.getElementById("distorted-visualization-canvas");
382   const transform = `rotate(${ window.window.rotations_euler[pos][2] + window.camera_offset[5] }deg)`;
383   canvas.style.transform = transform;
384   dist_canvas.style.transform = transform;
385   createLayers();
386 
387   window.deck.setProps({layers: window.layers});
388 }

479 /**
480  * One step in the animation of the train movement. Is tied to video frames as a callback for a new frame.
481  * Updates both the visualization and the GUI elements showing the progress.
482  * @param {DOMHighResTimeStamp } now - The time when the callback was called.
483  * @param {Object} metadata - An object defined in the documentation of requestVideoFrameCallback() method.
484  */
485 function animationStep(now, metadata) {
486   const video = document.getElementById('background-video');
487   if (window.animation_running) video.requestVideoFrameCallback(animationStep);
488   
489   // determine new position from video time and camera timestamps
490   if (metadata) {
491     while (window.camera_timestamps[window.position] < metadata.mediaTime) {
492       window.position += 1;
493     }
494   }
495 
496   if (window.position >= window.frames_cnt - 2) {  // end of animation
497     // stop the video
498     const video = document.getElementById('background-video');
499     video.pause();      // will fire stopDeckAnimation()
500     window.position = window.frames_cnt - 1;           // show the last frame
501   }
502 
503   // change point cloud data if we are not displaying the united point cloud
504   if (metadata && !window.display_united) {
505     // pcl_layers_positions is a circular buffer when the animation is running
506     while(window.pcl_timestamps[window.pcl_position] < metadata.mediaTime) {
507       // move to new position
508       window.pcl_position++;
509       // rewrite old data with new data
510       pcl_layers_positions[pcl_layers_index] = window.pcl_position;
511       // mark the next set of data as old data
512       pcl_layers_index = (pcl_layers_index + 1) % window.curr_pcl_layers_cnt;
513     }
514   }
515  
516   // update the visualization
517   window.updateDeck();
518   // update GUI elements
519   document.getElementById("camera-position-input").value = window.position;         // update input value
520   document.getElementById("camera-position-slider-input").value = window.position;  // update slider value
521   const time_sec = Math.floor(Math.max(video.currentTime - 0.001, 0)); // get time in seconds, round to whole number
522   const minutes = Math.floor(time_sec / 60);
523   const seconds = time_sec % 60;                                                    // update time label
524   document.getElementById("current-time-div").innerText = `${String(minutes).padStart(2, '0')}:${String(seconds).padStart(2, '0')}`;
525 }

527 /**
528  * Handles the beginning of the animation of the train movement.
529  */
530 function runDeckAnimation() {
531   if (window.position >= window.frames_cnt - 2) {  // it is at the end, start again from the beginning
532     const video = document.getElementById('background-video');
533     video.currentTime = 0;
534     window.position = 0;
535     window.pcl_position = 0;
536     pcl_layers_positions = new Array(window.curr_pcl_layers_cnt).fill(0);
537     window.updateDeck();
538   }
539 
540   // change play icon to pause icon
541   const icon = document.getElementById("play-button").querySelector("i");
542   icon.classList.remove("bi-play-fill");
543   icon.classList.add("bi-pause-fill");
544   
545   window.animation_running = true;
546   animationStep();
547 }

549 /**
550  * Handles the end of the animation of the train movement.
551  * Is either called from {@link playOrStop}, or is fired as a callback at the end of the animation.
552  */
553 function stopDeckAnimation() {
554   window.animation_running = false;
555   // change pause icon back to play icon
556   const icon = document.getElementById("play-button").querySelector("i");
557   icon.classList.add("bi-play-fill");
558   icon.classList.remove("bi-pause-fill");
559 
560   const video = document.getElementById('background-video');
561   // fix video offset
562   video.currentTime = video.currentTime;
563   // this has to be done with the GUI elements so that Dash knows about the changes
564   // could not be done while the animation was running, because it is slow 
565   dash_clientside.set_props("camera-position-input", {value: window.position});
566   dash_clientside.set_props("camera-position-slider-input", {value: window.position});
567   const time_sec = Math.floor(video.currentTime - 0.001); // get time in seconds, round to whole number
568   const minutes = Math.floor(time_sec / 60);
569   const seconds = time_sec % 60;
570   const label = `${String(minutes).padStart(2, '0')}:${String(seconds).padStart(2, '0')}`
571   dash_clientside.set_props("current-time-div", {children: label});
572 }

574 /**
575  * Starts/stops the animation of the train movement.
576  */
577 function playOrStop() {
578   if (!window.deck) return;
579 
580   const video = document.getElementById('background-video');
581   if (!window.animation_running) {
582     runDeckAnimation();        // run both deck animation and the video
583     video.play();
584     // define a callback that will run at the end of the animation
585     video.onpause = function(){ stopDeckAnimation() };
586   } else {
587     video.pause();
588   }
589 }

591 /**
592  * Changes the current position.
593  * @param {number} new_pos - The new position.
594  */
595 function jumpToPosition(new_pos) {
596   // update video
597   const video = document.getElementById('background-video');
598   const videoTime = window.camera_timestamps[new_pos];
599   video.currentTime = videoTime;
600   // update deck.gl visualization
601   window.position = new_pos;
602   changeLayersData();  // in case that we are not displaying united point cloud
603   updateDeck();  // call function defined in the JavaScript file
604   // update slider and input field and time label
605   dash_clientside.set_props("camera-position-slider-input", {value: new_pos});
606   dash_clientside.set_props("camera-position-input", {value: new_pos});
607   // update time label
608   const time_sec = Math.floor(videoTime);
609   const minutes = Math.floor(time_sec / 60);
610   const seconds = time_sec % 60;
611   const label = `${String(minutes).padStart(2, '0')}:${String(seconds).padStart(2, '0')}`
612   dash_clientside.set_props("current-time-div", {children: label});
613 }

615 /**
616  * Reacts to key presses.
617  *  - space - play/stop the animation
618  *  - right arrow/left arrow - jump 3 seconds forward/backwards
619  * @param {KeyboardEvent} e - The keypress event. 
620  */
621 function reactToKeyPress(e) {
622   if (e.key === ' ') {                   //  play/stop the animation on space press
623     playOrStop();
624 
625   } else if (e.key === 'ArrowLeft') {    // move the animation 3 seconds backwards on arrow left press
626     e.preventDefault();  // do not move other things 
627     // calculate the new time
628     let new_time = window.camera_timestamps[window.position] - 3;
629     if (new_time < 0) new_time = 0;
630     // find the position corresponding to the new time and jump to it
631     let new_pos = window.position;
632     while (window.camera_timestamps[new_pos] > new_time) {
633       new_pos -= 1;
634     }
635     jumpToPosition(new_pos);
636   } else if (e.key === 'ArrowRight') {    // move the animation 3 seconds forward on arrow right press
637     e.preventDefault();
638     // calculate the new time
639     let new_time = window.camera_timestamps[window.position] + 3;
640     const max_time = window.camera_timestamps[window.frames_cnt - 1];
641     if (new_time > max_time) new_time = max_time;
642     // find the position corresponding to the new time and jump to it
643     let new_pos = window.position;
644     while (window.camera_timestamps[new_pos] < new_time) {
645       new_pos += 1;
646     }
647     jumpToPosition(new_pos);
648   } else if (e.key === 'ArrowUp' || e.key === 'ArrowDown') {   // do not move camera sliders by arrows
649     if (document.activeElement.id === 'camera-x-slider-input' 
650       || document.activeElement.id === 'camera-y-slider-input'
651       || document.activeElement.id === 'camera-z-slider-input'
652       || document.activeElement.id === 'camera-yaw-slider-input'
653       || document.activeElement.id === 'camera-pitch-slider-input'
654       || document.activeElement.id === 'camera-roll-slider-input') e.preventDefault();
655   }
656 }
657 
658 // set the callback to the key press event
659 document.addEventListener('keydown', reactToKeyPress);

661 // a special accomodation for Google Chrome, to make the animation continue running when
662 // the user switches to another window in the browser and then switches back to this window
663 // (Chrome stops the animation when the window is not visible)
664 document.addEventListener("visibilitychange", () => {
665   // wait 40 ms, because this runs before Chrome makes the animation run again
666   setTimeout(() => {
667     const video = document.getElementById('background-video');
668     if (document.visibilityState === "visible" && !video.paused && !video.ended) {
669       runDeckAnimation();
670     }
671   }, 40);
672 });
